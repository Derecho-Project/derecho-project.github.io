61 void SST<DerivedSST>::detect() {
62  pthread_setname_np(pthread_self(), "sst_detect");
63  if(!thread_start) {
64  std::unique_lock<std::mutex> lock(thread_start_mutex);
65  thread_start_cv.wait(lock, [this]() { return thread_start; });
66  }
67  struct timespec last_time, cur_time;
68  clock_gettime(CLOCK_REALTIME, &last_time);
69 
70  while(!thread_shutdown) {
71  try {
72  bool predicate_fired = false;
73  // Take the predicate lock before reading the predicate lists
74  std::unique_lock<std::mutex> predicates_lock(predicates.predicate_mutex);
75 
76  // one time predicates need to be evaluated only until they become true
77  for(auto& pred : predicates.one_time_predicates) {
78  if(pred != nullptr && (pred->first(*derived_this) == true)) {
79  predicate_fired = true;
80  // Copy the trigger pointer locally, so it can continue running without
81  // segfaulting even if this predicate gets deleted when we unlock predicates_lock
82  std::shared_ptr<typename Predicates<DerivedSST>::trig> trigger(pred->second);
83  predicates_lock.unlock();
84  (*trigger)(*derived_this);
85  predicates_lock.lock();
86  // erase the predicate as it was just found to be true
87  pred.reset();
88  }
89  }
90 
91  // recurrent predicates are evaluated each time they are found to be true
92  for(auto& pred : predicates.recurrent_predicates) {
93  if(pred != nullptr && (pred->first(*derived_this) == true)) {
94  predicate_fired = true;
95  std::shared_ptr<typename Predicates<DerivedSST>::trig> trigger(pred->second);
96  predicates_lock.unlock();
97  (*trigger)(*derived_this);
98  predicates_lock.lock();
99  }
100  }
101 
102  // transition predicates are only evaluated when they change from false to true
103  // We need to use iterators here because we need to iterate over two lists in parallel
104  auto pred_it = predicates.transition_predicates.begin();
105  auto pred_state_it = predicates.transition_predicate_states.begin();
106  while(pred_it != predicates.transition_predicates.end()) {
107  if(*pred_it != nullptr) {
108  //*pred_state_it is the previous state of the predicate at *pred_it
109  bool curr_pred_state = (*pred_it)->first(*derived_this);
110  if(curr_pred_state == true && *pred_state_it == false) {
111  predicate_fired = true;
112  std::shared_ptr<typename Predicates<DerivedSST>::trig> trigger(
113  (*pred_it)->second);
114  predicates_lock.unlock();
115  (*trigger)(*derived_this);
116  predicates_lock.lock();
117  }
118  *pred_state_it = curr_pred_state;
119 
120  ++pred_it;
121  ++pred_state_it;
122  }
123  }
124 
125  if(predicate_fired) {
126  // update last time
127  clock_gettime(CLOCK_REALTIME, &last_time);
128  } else {
129  clock_gettime(CLOCK_REALTIME, &cur_time);
130  // check if the system has been inactive for enough time to induce sleep
131  double time_elapsed_in_ms = (cur_time.tv_sec - last_time.tv_sec) * 1e3
132  + (cur_time.tv_nsec - last_time.tv_nsec) / 1e6;
133  if(time_elapsed_in_ms > 1) {
134  predicates_lock.unlock();
135  using namespace std::chrono_literals;
136  std::this_thread::sleep_for(1ms);
137  predicates_lock.lock();
138  }
139  }
140  //Still to do: Clean up deleted predicates
141  } catch(const std::exception& e) {
142  std::cout << "Exception in the SST detect thread: " << e.what() << std::endl;
143  }
144  }
