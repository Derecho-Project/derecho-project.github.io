242

244  dbg_default_trace("connect to remote node.");
245  ssize_t nRead;
246  struct fi_eq_cm_entry entry;
247  uint32_t event;
248

272  dbg_default_trace("initiating a connection.\n");

290  if (nRead != sizeof(entry)) {
291  dbg_default_error("failed to connect remote.");
292  CRASH_WITH_MESSAGE("failed to connect remote. nRead=%ld.\n",nRead);
293  }

303  fi_freeinfo(client_info);
304  }

593  std::pair<uint32_t, std::pair<int32_t, int32_t>> lf_poll_completion() {
594  struct fi_cq_entry entry;
595  int poll_result;
596 
597  while(!shutdown) {
598  poll_result = 0;
599  for(int i = 0; i < 50; ++i) {
600  poll_result = fi_cq_read(g_ctxt.cq, &entry, 1);
601  if(poll_result && (poll_result!=-FI_EAGAIN)) {
602  break;
603  }
604  }
605  if(poll_result && (poll_result!=-FI_EAGAIN)) {
606  break;
607  }
608  // util::polling_data.wait_for_requests();
609  }
610  // not sure what to do when we cannot read entries off the CQ
611  // this means that something is wrong with the local node
612  if((poll_result < 0) && (poll_result != -FI_EAGAIN)) {
613  struct fi_cq_err_entry eentry;
614  fi_cq_readerr(g_ctxt.cq, &eentry, 0);
615

622 #endif
623  dbg_default_error("\top_context:ce_idx={},remote_id={}",sctxt->ce_idx,sctxt->remote_id);

625 #ifdef DEBUG_FOR_RELEASE

631  printf("\tolen=0x%x\n",eentry.olen);

633 #endif//DEBUG_FOR_RELEASE

654 #endif//DEBUG_FOR_RELEASE

656  struct lf_sender_ctxt * sctxt = (struct lf_sender_ctxt *)eentry.op_context;
657  return {sctxt->ce_idx, {sctxt->remote_id, -1}};
658  } else {
659  dbg_default_error("\tFailed polling the completion queue");
660  fprintf(stderr,"Failed polling the completion queue");
661  return {(uint32_t)0xFFFFFFFF,{0,-1}}; // we don't know who sent the message.
662  // CRASH_WITH_MESSAGE("failed polling the completion queue");
663  }
664  }
665  if (!shutdown) {
666  struct lf_sender_ctxt * sctxt = (struct lf_sender_ctxt *)entry.op_context;
667  if (sctxt == NULL) {
668  dbg_default_debug("WEIRD: we get an entry with op_context = NULL.");
669  return {0xFFFFFFFFu,{0,0}}; // return a bad entry: weird!!!!
670  } else {
671  dbg_default_trace("Normal: we get an entry with op_context = {}.",(long long unsigned)sctxt);
672  return {sctxt->ce_idx, {sctxt->remote_id, 1}};
673  }
674  } else { // shutdown return a bad entry
675  return {0,{0,0}};
676  }
