13 void RestartState::load_ragged_trim(const View& curr_view) {
14  whenlog(auto logger = LoggerFactory::getDefaultLogger(););
15  /* Iterate through all subgroups by type, rather than iterating through my_subgroups,
16  * so that I have access to the type ID. This wastes time, but I don't have a map
17  * from subgroup ID to subgroup_type_id within curr_view. */
18  for(const auto& type_id_and_indices : curr_view.subgroup_ids_by_type_id) {
19  for(uint32_t subgroup_index = 0; subgroup_index < type_id_and_indices.second.size(); ++subgroup_index) {
20  subgroup_id_t subgroup_id = type_id_and_indices.second.at(subgroup_index);
21  //We only care if the subgroup's ID is in my_subgroups
22  auto subgroup_shard_ptr = curr_view.my_subgroups.find(subgroup_id);
23  if(subgroup_shard_ptr != curr_view.my_subgroups.end()) {
24  //If the subgroup ID is in my_subgroups, its value is this node's shard number
25  uint32_t shard_num = subgroup_shard_ptr->second;
26  std::unique_ptr<RaggedTrim> ragged_trim = persistent::loadObject<RaggedTrim>(
27  ragged_trim_filename(subgroup_id, shard_num).c_str());
28  //If there was a logged ragged trim from an obsolete View, it's the same as not having a logged ragged trim
29  if(ragged_trim == nullptr || ragged_trim->vid < curr_view.vid) {
30  whenlog(logger->debug("No ragged trim information found for subgroup {}, synthesizing it from logs", subgroup_id););
31  //Get the latest persisted version number from this subgroup's object's log
32  //(this requires converting the type ID to a std::type_index
33  persistent::version_t last_persisted_version = persistent::getMinimumLatestPersistedVersion(curr_view.subgroup_type_order.at(type_id_and_indices.first),
34  subgroup_index, shard_num);
35  int32_t last_vid, last_seq_num;
36  std::tie(last_vid, last_seq_num) = persistent::unpack_version<int32_t>(last_persisted_version);
37  //Divide the sequence number into sender rank and message counter
38  uint32_t num_shard_senders = curr_view.subgroup_shard_views.at(subgroup_id).at(shard_num).num_senders();
39  int32_t last_message_counter = last_seq_num / num_shard_senders;
40  uint32_t last_sender = last_seq_num % num_shard_senders;
41  /* Fill max_received_by_sender: In round-robin order, all senders ranked below
42  * the last sender delivered last_message_counter, while all senders ranked above
43  * the last sender have only delivered last_message_counter-1. */
44  std::vector<int32_t> max_received_by_sender(num_shard_senders);
45  for(uint sender_rank = 0; sender_rank <= last_sender; ++sender_rank) {
46  max_received_by_sender[sender_rank] = last_message_counter;
47  }
48  for(uint sender_rank = last_sender + 1; sender_rank < num_shard_senders; ++sender_rank) {
49  max_received_by_sender[sender_rank] = last_message_counter - 1;
50  }
51  ragged_trim = std::make_unique<RaggedTrim>(subgroup_id, shard_num, last_vid, -1, max_received_by_sender);
52  }
53  //operator[] is intentional: default-construct an inner std::map at subgroup_id
54  //Note that the inner map will only one entry, except on the restart leader where it will have one for every shard
55  logged_ragged_trim[subgroup_id].emplace(shard_num, std::move(ragged_trim));
56  } // if(subgroup_shard_ptr != curr_view->my_subgroups.end())
57  } // for(subgroup_index)
58  }

61 persistent::version_t RestartState::ragged_trim_to_latest_version(const int32_t view_id,
62  const std::vector<int32_t>& max_received_by_sender) {
63  uint32_t num_shard_senders = max_received_by_sender.size();
64  //Determine the last deliverable sequence number using the same logic as deliver_messages_upto
65  int32_t max_seq_num = 0;
66  for(uint sender = 0; sender < num_shard_senders; sender++) {

71  return persistent::combine_int32s(view_id, max_seq_num);

105 void RestartLeaderState::await_quorum(tcp::connection_listener& server_socket) {
106  bool ready_to_restart = false;
107  int time_remaining_ms = RESTART_LEADER_TIMEOUT;
108  while(time_remaining_ms > 0) {
109  using namespace std::chrono;
110  auto start_time = high_resolution_clock::now();
111  std::optional<tcp::socket> client_socket = server_socket.try_accept(time_remaining_ms);
112  auto end_time = high_resolution_clock::now();
113  milliseconds time_waited = duration_cast<milliseconds>(end_time - start_time);
114  time_remaining_ms -= time_waited.count();
115  if(client_socket) {
116  node_id_t joiner_id = 0;
117  client_socket->read(joiner_id);
118  client_socket->write(JoinResponse{JoinResponseCode::TOTAL_RESTART, my_id});
119  whenlog(logger->debug("Node {} rejoined", joiner_id););
120  rejoined_node_ids.emplace(joiner_id);
121 
122  //Receive and process the joining node's logs of the last known View and RaggedTrim
123  receive_joiner_logs(joiner_id, *client_socket);
124 
125  //Receive the joining node's ports - this is part of the standard join logic
126  uint16_t joiner_gms_port = 0;
127  client_socket->read(joiner_gms_port);
128  uint16_t joiner_rpc_port = 0;
129  client_socket->read(joiner_rpc_port);
130  uint16_t joiner_sst_port = 0;
131  client_socket->read(joiner_sst_port);
132  uint16_t joiner_rdmc_port = 0;
133  client_socket->read(joiner_rdmc_port);
134  const ip_addr_t& joiner_ip = client_socket->get_remote_ip();
135  rejoined_node_ips_and_ports[joiner_id] = {joiner_ip, joiner_gms_port,
136  joiner_rpc_port, joiner_sst_port, joiner_rdmc_port};
137  //Done receiving from this socket (for now), so store it in waiting_join_sockets for later
138  waiting_join_sockets.emplace(joiner_id, std::move(*client_socket));
139  //Check for quorum
140  ready_to_restart = has_restart_quorum();
141  //If all the members have rejoined, no need to keep waiting
142  if(std::includes(rejoined_node_ids.begin(), rejoined_node_ids.end(),
143  last_known_view_members.begin(), last_known_view_members.end())) {
144  return;
145  }
146  } else if(!ready_to_restart) {
147  //Accept timed out, but we haven't heard from enough nodes yet, so reset the timer
148  time_remaining_ms = RESTART_LEADER_TIMEOUT;
149  }
150  }

153 bool RestartLeaderState::has_restart_quorum() {
154  //Compute rejoined_node_ids.intersect(last_known_view_members)
155  //but with a lot of unnecessary repetitive boilerplate because it's the STL
156  std::set<node_id_t> intersection_of_ids;
157  std::set_intersection(rejoined_node_ids.begin(), rejoined_node_ids.end(),
158  last_known_view_members.begin(), last_known_view_members.end(),
159  std::inserter(intersection_of_ids, intersection_of_ids.end()));
160  if(intersection_of_ids.size() < (last_known_view_members.size() / 2) + 1 || !contains_at_least_one_member_per_subgroup(rejoined_node_ids, *curr_view)) {
161  return false;
162  }
163  //If we have a sufficient number of members, attempt to compute a restart view
164  //If that fails, we know we don't have a quorum, but if it succeeds, we
165  //both have a quorum and know the restart view
166  return compute_restart_view();

169 void RestartLeaderState::receive_joiner_logs(const node_id_t& joiner_id, tcp::socket& client_socket) {
170  //Receive the joining node's saved View
171  std::size_t size_of_view;
172  client_socket.read(size_of_view);
173  char view_buffer[size_of_view];
174  client_socket.read(view_buffer, size_of_view);
175  std::unique_ptr<View> client_view = mutils::from_bytes<View>(nullptr, view_buffer);
176 
177  if(client_view->vid > curr_view->vid) {
178  whenlog(logger->trace("Node {} had newer view {}, replacing view {} and discarding ragged trim", joiner_id, client_view->vid, curr_view->vid););
179  //The joining node has a newer View, so discard any ragged trims that are not longest-log records
180  for(auto& subgroup_to_map : restart_state.logged_ragged_trim) {
181  auto trim_map_iterator = subgroup_to_map.second.begin();
182  while(trim_map_iterator != subgroup_to_map.second.end()) {
183  if(trim_map_iterator->second->leader_id != -1) {
184  trim_map_iterator = subgroup_to_map.second.erase(trim_map_iterator);
185  } else {
186  ++trim_map_iterator;
187  }
188  }
189  }
190  }
191  //Receive the joining node's RaggedTrims
192  std::size_t num_of_ragged_trims;
193  client_socket.read(num_of_ragged_trims);
194  for(std::size_t i = 0; i < num_of_ragged_trims; ++i) {
195  std::size_t size_of_ragged_trim;
196  client_socket.read(size_of_ragged_trim);
197  char buffer[size_of_ragged_trim];
198  client_socket.read(buffer, size_of_ragged_trim);
199  std::unique_ptr<RaggedTrim> ragged_trim = mutils::from_bytes<RaggedTrim>(nullptr, buffer);
200  whenlog(logger->trace("Received ragged trim for subgroup {}, shard {} from node {}", ragged_trim->subgroup_id, ragged_trim->shard_num, joiner_id););
201  /* If the joining node has an obsolete View, we only care about the
202  * "ragged trims" if they are actually longest-log records and from
203  * a newer view than any ragged trims we have for this subgroup. */
204  if(client_view->vid < curr_view->vid && ragged_trim->leader_id != -1) { //-1 means the RaggedTrim is a log report
205  continue;
206  }
207  /* Determine if this node might end up being the "restart leader" for its subgroup
208  * because it has the longest log. Note that comparing log versions implicitly
209  * compares VIDs, so a ragged trim from a newer View is always "longer" */
210  persistent::version_t ragged_trim_log_version = RestartState::ragged_trim_to_latest_version(ragged_trim->vid, ragged_trim->max_received_by_sender);
211  if(ragged_trim_log_version > longest_log_versions[ragged_trim->subgroup_id][ragged_trim->shard_num]) {
212  whenlog(logger->trace("Latest logged persistent version for subgroup {}, shard {} is now {}, which is at node {}", ragged_trim->subgroup_id, ragged_trim->shard_num, ragged_trim_log_version, joiner_id););
213  longest_log_versions[ragged_trim->subgroup_id][ragged_trim->shard_num] = ragged_trim_log_version;
214  nodes_with_longest_log[ragged_trim->subgroup_id][ragged_trim->shard_num] = joiner_id;
215  }
216  if(client_view->vid <= curr_view->vid) {
217  //In both of these cases, only keep the ragged trim if it is newer than anything we have
218  auto existing_ragged_trim = restart_state.logged_ragged_trim[ragged_trim->subgroup_id].find(ragged_trim->shard_num);
219  if(existing_ragged_trim == restart_state.logged_ragged_trim[ragged_trim->subgroup_id].end()) {
220  whenlog(logger->trace("Adding node {}'s ragged trim to map, because we don't have one for shard ({}, {})", joiner_id, ragged_trim->subgroup_id, ragged_trim->shard_num););
221  //operator[] is intentional: Default-construct an inner std::map if one doesn't exist at this ID
222  restart_state.logged_ragged_trim[ragged_trim->subgroup_id].emplace(ragged_trim->shard_num,
223  std::move(ragged_trim));
224  } else if(existing_ragged_trim->second->vid <= ragged_trim->vid) {
225  existing_ragged_trim->second = std::move(ragged_trim);
226  }
227  } else {
228  //The client had a newer View, so accept everything it sends
229  restart_state.logged_ragged_trim[ragged_trim->subgroup_id].emplace(ragged_trim->shard_num,
230  std::move(ragged_trim));
231  }
232  }
233  //Replace curr_view if the client's view was newer
234  if(client_view->vid > curr_view->vid) {
235  client_view->subgroup_type_order = curr_view->subgroup_type_order;
236  curr_view.swap(client_view);
237  //Remake the std::set version of curr_view->members
238  last_known_view_members.clear();
239  last_known_view_members.insert(curr_view->members.begin(), curr_view->members.end());
240  }
241 }

249 int64_t RestartLeaderState::send_restart_view() {
250  members_sent_restart_view.clear();
251  for(auto waiting_sockets_iter = waiting_join_sockets.begin();
252  waiting_sockets_iter != waiting_join_sockets.end();) {
253  std::size_t view_buffer_size = mutils::bytes_size(*restart_view);
254  std::size_t params_buffer_size = mutils::bytes_size(derecho_params);
255  std::size_t leaders_buffer_size = mutils::bytes_size(nodes_with_longest_log);
256  char view_buffer[view_buffer_size];
257  char params_buffer[params_buffer_size];
258  char leaders_buffer[leaders_buffer_size];
259  bool send_success;
260  //Within this try block, any send that returns failure throws the ID of the node that failed
261  try {
262  whenlog(logger->debug("Sending post-recovery view {} to node {}", restart_view->vid, waiting_sockets_iter->first););
263  send_success = waiting_sockets_iter->second.write(view_buffer_size);
264  if(!send_success) {
265  throw waiting_sockets_iter->first;
266  }
267  mutils::to_bytes(*restart_view, view_buffer);
268  send_success = waiting_sockets_iter->second.write(view_buffer, view_buffer_size);
269  if(!send_success) {
270  throw waiting_sockets_iter->first;
271  }
272  send_success = waiting_sockets_iter->second.write(params_buffer_size);
273  if(!send_success) {
274  throw waiting_sockets_iter->first;
275  }
276  mutils::to_bytes(derecho_params, params_buffer);
277  send_success = waiting_sockets_iter->second.write(params_buffer, params_buffer_size);
278  if(!send_success) {
279  throw waiting_sockets_iter->first;
280  }
281  whenlog(logger->debug("Sending ragged-trim information to node {}", waiting_sockets_iter->first););
282  std::size_t num_ragged_trims = multimap_size(restart_state.logged_ragged_trim);
283  send_success = waiting_sockets_iter->second.write(num_ragged_trims);
284  if(!send_success) {
285  throw waiting_sockets_iter->first;
286  }
287  //Unroll the maps and send each RaggedTrim individually, since it contains its subgroup_id and shard_num
288  for(const auto& subgroup_to_shard_map : restart_state.logged_ragged_trim) {
289  for(const auto& shard_trim_pair : subgroup_to_shard_map.second) {
290  std::size_t trim_buffer_size = mutils::bytes_size(*shard_trim_pair.second);
291  char trim_buffer[trim_buffer_size];
292  send_success = waiting_sockets_iter->second.write(trim_buffer_size);
293  if(!send_success) {
294  throw waiting_sockets_iter->first;
295  }
296  mutils::to_bytes(*shard_trim_pair.second, trim_buffer);
297  send_success = waiting_sockets_iter->second.write(trim_buffer, trim_buffer_size);
298  if(!send_success) {
299  throw waiting_sockets_iter->first;
300  }
301  }
302  }
303  whenlog(logger->debug("Sending longest-log locations to node {}", waiting_sockets_iter->first););
304  send_success = waiting_sockets_iter->second.write(leaders_buffer_size);
305  if(!send_success) {
306  throw waiting_sockets_iter->first;
307  }
308  mutils::to_bytes(nodes_with_longest_log, leaders_buffer);
309  send_success = waiting_sockets_iter->second.write(leaders_buffer, leaders_buffer_size);
310  if(!send_success) {
311  throw waiting_sockets_iter->first;
312  }
313  members_sent_restart_view.emplace(waiting_sockets_iter->first);
314  waiting_sockets_iter++;
315  } catch(node_id_t failed_node) {
316  //All send failures will end up here.
317  //Close the failed socket, delete it from rejoined_node_ids, and return the ID of the failed node.
318  waiting_join_sockets.erase(waiting_sockets_iter);
319  rejoined_node_ips_and_ports.erase(failed_node);
320  rejoined_node_ids.erase(failed_node);
321  return failed_node;
322  }
323  } //for (waiting_join_sockets)
324 
325  //Save this to a class member so that we still have it in send_objects_if_total_restart()
326  restart_state.restart_shard_leaders = nodes_with_longest_log;
327  //Return -1 to indicate success: no node failed.
328  return -1;
329 }

401 std::unique_ptr<View> RestartLeaderState::update_curr_and_next_restart_view() {
402  //Nodes that were not in the last view but have restarted will immediately "join" in the new view
403  std::vector<node_id_t> nodes_to_add_in_next_view;
404  std::vector<std::tuple<ip_addr_t, uint16_t, uint16_t, uint16_t, uint16_t>> ips_and_ports_to_add_in_next_view;
405  for(const auto& id_socket_pair : waiting_join_sockets) {
406  node_id_t joiner_id = id_socket_pair.first;
407  int joiner_rank = curr_view->rank_of(joiner_id);
408  if(joiner_rank == -1) {
409  nodes_to_add_in_next_view.emplace_back(joiner_id);
410  ips_and_ports_to_add_in_next_view.emplace_back(rejoined_node_ips_and_ports.at(joiner_id));
411  //If this node had been marked as failed, but was still in the view, un-fail it
412  } else if(curr_view->failed[joiner_rank] == true) {
413  curr_view->failed[joiner_rank] = false;
414  curr_view->num_failed--;
415  }
416  }
417  //Mark any nodes from the last view that haven't yet responded as failed
418  for(std::size_t rank = 0; rank < curr_view->members.size(); ++rank) {
419  if(rejoined_node_ids.count(curr_view->members[rank]) == 0
420  && !curr_view->failed[rank]) {
421  curr_view->failed[rank] = true;
422  curr_view->num_failed++;
423  }
424  }

430 std::unique_ptr<View> RestartLeaderState::make_next_view(const std::unique_ptr<View>& curr_view,
431  const std::vector<node_id_t>& joiner_ids,
432  const std::vector<std::tuple<ip_addr_t, uint16_t, uint16_t, uint16_t, uint16_t>>& joiner_ips_and_ports
433  whenlog(, std::shared_ptr<spdlog::logger> logger)) {
434  int next_num_members = curr_view->num_members - curr_view->num_failed + joiner_ids.size();
435  std::vector<node_id_t> members(next_num_members), departed;
436  std::vector<char> failed(next_num_members);
437  std::vector<std::tuple<ip_addr_t, uint16_t, uint16_t, uint16_t, uint16_t>> member_ips_and_ports(next_num_members);
438  int next_unassigned_rank = curr_view->next_unassigned_rank;
439  std::set<int> leave_ranks;
440  for(std::size_t rank = 0; rank < curr_view->failed.size(); ++rank) {
441  if(curr_view->failed[rank]) {
442  leave_ranks.emplace(rank);
443  }
444  }
445  for(std::size_t i = 0; i < joiner_ids.size(); ++i) {
446  int new_member_rank = curr_view->num_members - leave_ranks.size() + i;
447  members[new_member_rank] = joiner_ids[i];
448  member_ips_and_ports[new_member_rank] = joiner_ips_and_ports[i];
449  whenlog(logger->debug("Restarted next view will add new member with id {}", joiner_ids[i]););
450  }
451  for(const auto& leaver_rank : leave_ranks) {
452  departed.emplace_back(curr_view->members[leaver_rank]);
453  //Decrement next_unassigned_rank for every failure, unless the failure wasn't assigned to a subgroup anyway
454  if(leaver_rank <= curr_view->next_unassigned_rank) {
455  next_unassigned_rank--;
456  }
457  }
458  whenlog(logger->debug("Next view will exclude {} failed members.", leave_ranks.size()););
459  //Copy member information, excluding the members that have failed
460  int new_rank = 0;
461  for(int old_rank = 0; old_rank < curr_view->num_members; ++old_rank) {
462  //This is why leave_ranks needs to be a set
463  if(leave_ranks.find(old_rank) == leave_ranks.end()) {
464  members[new_rank] = curr_view->members[old_rank];
465  member_ips_and_ports[new_rank] = curr_view->member_ips_and_ports[old_rank];
466  failed[new_rank] = curr_view->failed[old_rank];
467  ++new_rank;
468  }
469  }
470 
471  //Initialize my_rank in next_view
472  int32_t my_new_rank = -1;
473  node_id_t myID = curr_view->members[curr_view->my_rank];
474  for(int i = 0; i < next_num_members; ++i) {
475  if(members[i] == myID) {
476  my_new_rank = i;
477  break;
478  }
479  }
480  if(my_new_rank == -1) {
481  whenlog(logger->flush(););
482  throw derecho_exception("Recovery leader wasn't in the next view it computed?!?!");
483  }
484 
485  auto next_view = std::make_unique<View>(curr_view->vid + 1, members, member_ips_and_ports, failed,
486  joiner_ids, departed, my_new_rank, next_unassigned_rank,
487  curr_view->subgroup_type_order);
488  next_view->i_know_i_am_leader = curr_view->i_know_i_am_leader;
489  return std::move(next_view);
490 }

500  }
501  }

503  return false;
504  }
505  }
506  }
