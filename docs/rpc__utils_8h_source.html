258  void fulfill_map(const node_list_t& who) {
259  map_fulfilled = true;
260  std::unique_ptr<reply_map<Ret>> to_add = std::make_unique<reply_map<Ret>>();
261  for(const auto& e : who) {
262  to_add->emplace(e, populated_promises[e].get_future());
263  }
264  dest_nodes.insert(who.begin(), who.end());
265  pending_map.set_value(std::move(to_add));
266  }
267 
268  void set_exception_for_removed_node(const node_id_t& removed_nid) {
269  assert(map_fulfilled);
270  if(dest_nodes.find(removed_nid) != dest_nodes.end()
271  && responded_nodes.find(removed_nid) == responded_nodes.end()) {
272  set_exception(removed_nid,
273  std::make_exception_ptr(
274  node_removed_from_group_exception{removed_nid}));
275  }
276  }
277 
278  void set_value(const node_id_t& nid, const Ret& v) {
279  responded_nodes.insert(nid);
