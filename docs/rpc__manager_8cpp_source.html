72 void RPCManager::rpc_message_handler(subgroup_id_t subgroup_id, node_id_t sender_id, char* msg_buf, uint32_t payload_size) {
73  // WARNING: This assumes the current view doesn't change during execution! (It accesses curr_view without a lock).
74  // extract the destination vector
75  size_t dest_size = ((size_t*)msg_buf)[0];
76  msg_buf += sizeof(size_t);
77  payload_size -= sizeof(size_t);
78  bool in_dest = false;
79  for(size_t i = 0; i < dest_size; ++i) {
80  auto n = ((node_id_t*)msg_buf)[0];
81  msg_buf += sizeof(node_id_t);
82  payload_size -= sizeof(node_id_t);
83  if(n == nid) {
84  in_dest = true;
85  }
86  }
87  if(in_dest || dest_size == 0) {
88  //Use the reply-buffer allocation lambda to detect whether handle_receive generated a reply
89  size_t reply_size = 0;
90  char* reply_buf;
91  parse_and_receive(msg_buf, payload_size, [this, &reply_buf, &reply_size, &sender_id](size_t size) -> char* {
92  reply_size = size;
93  if(reply_size <= connections->get_max_p2p_size()) {
94  reply_buf = (char*)connections->get_sendbuffer_ptr(
95  connections->get_node_rank(sender_id), sst::REQUEST_TYPE::RPC_REPLY);
96  return reply_buf;
97  } else {
98  // the reply size is too large - not part of the design to handle it
99  return nullptr;
100  }
101  });
102  if(reply_size > 0) {
103  if(sender_id == nid) {
104  //The RPC message expects replies, and I was the sender, so I might have a reply-map that needs fulfilling
105  if(dest_size == 0) {
106  //Destination was "all nodes in my shard of the subgroup"
107  int my_shard = view_manager.curr_view->multicast_group->get_subgroup_settings().at(subgroup_id).shard_num;
108  std::lock_guard<std::mutex> lock(pending_results_mutex);
109  assert(!toFulfillQueue.empty());
110  // whenlog(logger->trace("Calling fulfill_map on toFulfillQueue.front(), its size is {}", toFulfillQueue.size());)
111  toFulfillQueue.front().get().fulfill_map(
112  view_manager.curr_view->subgroup_shard_views.at(subgroup_id).at(my_shard).members);
113  fulfilledList.push_back(std::move(toFulfillQueue.front()));
114  toFulfillQueue.pop();
115 // whenlog(logger->trace("Popped a PendingResults from toFulfillQueue, size is now {}", toFulfillQueue.size());)
116  }
117  //Immediately handle the reply to myself
118  parse_and_receive(
119  reply_buf, reply_size,
120  [](size_t size) -> char* { assert_always(false); });
121  } else {
122  connections->send(connections->get_node_rank(sender_id));
123  }
124  } else {
125  whenlog(logger->trace("RPC message handled, no reply necessary.");)
126  if(sender_id == nid && dest_size == 0) {
127  std::lock_guard<std::mutex> lock(pending_results_mutex);
128  assert(!toFulfillQueue.empty());
129  toFulfillQueue.pop();
130 // whenlog(logger->trace("Deleted a useless PendingResults from toFulfillQueue, size is now {}", toFulfillQueue.size());)
131  }
