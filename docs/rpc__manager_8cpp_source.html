67 void RPCManager::rpc_message_handler(subgroup_id_t subgroup_id, node_id_t sender_id, char* msg_buf, uint32_t payload_size) {
68  // WARNING: This assumes the current view doesn't change during execution! (It accesses curr_view without a lock).
69  // extract the destination vector
70  size_t dest_size = ((size_t*)msg_buf)[0];
71  msg_buf += sizeof(size_t);
72  payload_size -= sizeof(size_t);
73  bool in_dest = false;
74  for(size_t i = 0; i < dest_size; ++i) {
75  auto n = ((node_id_t*)msg_buf)[0];
76  msg_buf += sizeof(node_id_t);
77  payload_size -= sizeof(node_id_t);
78  if(n == nid) {
79  in_dest = true;
80  }
81  }
82  if(in_dest || dest_size == 0) {
83  //Use the reply-buffer allocation lambda to detect whether parse_and_receive generated a reply
84  size_t reply_size = 0;
85  char* reply_buf;
86  parse_and_receive(msg_buf, payload_size, [this, &reply_buf, &reply_size, &sender_id](size_t size) -> char* {
87  reply_size = size;
88  if(reply_size <= connections->get_max_p2p_size()) {
89  reply_buf = (char*)connections->get_sendbuffer_ptr(
90  connections->get_node_rank(sender_id), sst::REQUEST_TYPE::RPC_REPLY);
91  return reply_buf;
92  } else {
93  // the reply size is too large - not part of the design to handle it
94  return nullptr;
95  }
96  });
97  if(sender_id == nid) {
98  //This is a self-receive of an RPC message I sent, so I have a reply-map that needs fulfilling
99  int my_shard = view_manager.curr_view->multicast_group->get_subgroup_settings().at(subgroup_id).shard_num;
100  std::unique_lock<std::mutex> lock(pending_results_mutex);
101  // because of a race condition, toFulfillQueue can genuinely be empty
102  // so we shouldn't assert that it is empty
103  // instead we should sleep on a condition variable and let the main thread that called the orderedSend signal us
104  // although the race condition is infinitely rare
105  pending_results_cv.wait(lock, [&]() { return !toFulfillQueue.empty(); });
106  // whenlog(logger->trace("Calling fulfill_map on toFulfillQueue.front(), its size is {}", toFulfillQueue.size());)
107  //We now know the membership of "all nodes in my shard of the subgroup" in the current view
108  toFulfillQueue.front().get().fulfill_map(
109  view_manager.curr_view->subgroup_shard_views.at(subgroup_id).at(my_shard).members);
110  fulfilledList.push_back(std::move(toFulfillQueue.front()));
111  toFulfillQueue.pop();
112  // whenlog(logger->trace("Popped a PendingResults from toFulfillQueue, size is now {}", toFulfillQueue.size());)
113  if(reply_size > 0) {
114  //Since this was a self-receive, the reply also goes to myself
115  parse_and_receive(
116  reply_buf, reply_size,
117  [](size_t size) -> char* { assert_always(false); });
118  }
119  } else if(reply_size > 0) {
120  //Otherwise, the only thing to do is send the reply (if there was one)
121  connections->send(connections->get_node_rank(sender_id));
122  }
123  }
124 }
