265  const auto tid = std::this_thread::get_id();
266  // get id first
267  uint32_t ce_idx = util::polling_data.get_index(tid);
268  while(!thread_shutdown) {
269  std::this_thread::sleep_for(std::chrono::milliseconds(1));
270  if(num_rdma_writes < 1000) {
271  continue;
272  }
273  num_rdma_writes = 0;
274 
275  util::polling_data.set_waiting(tid);
276 #ifdef USE_VERBS_API
277  struct verbs_sender_ctxt sctxt[num_members];
278 #else
279  struct lf_sender_ctxt sctxt[num_members];
280 #endif
281 
282  for(uint rank = 0; rank < num_members; ++rank) {
283  if(rank == my_index) {
284  continue;
285  }
286 
287  sctxt[rank].remote_id = rank;
288  sctxt[rank].ce_idx = ce_idx;
289 
290  res_vec[rank]->post_remote_write_with_completion(&sctxt[rank], p2p_buf_size - sizeof(bool), sizeof(bool));
291  }

295  unsigned long start_time_msec;
296  unsigned long cur_time_msec;
297  struct timeval cur_time;
298 
299  // wait for completion for a while before giving up of doing it ..
300  gettimeofday(&cur_time, NULL);
301  start_time_msec = (cur_time.tv_sec * 1000) + (cur_time.tv_usec / 1000);
302 
303  uint32_t num_completions = 0;

305  std::optional<std::pair<int32_t, int32_t>> ce;
306  while(true) {
