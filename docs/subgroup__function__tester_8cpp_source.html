145  std::cout << "Subgroup type " << subgroup_type.name() << " failed to provision, marking View inadequate" << std::endl

187 std::unique_ptr<View> make_next_view(const View& curr_view,
188  const std::set<int>& leave_ranks,
189  const std::vector<node_id_t>& joiner_ids,
190  const std::vector<std::tuple<ip_addr_t, uint16_t, uint16_t, uint16_t, uint16_t>>& joiner_ips_and_ports) {
191  int next_num_members = curr_view.num_members - leave_ranks.size() + joiner_ids.size();
192  std::vector<node_id_t> joined, members(next_num_members), departed;
193  std::vector<char> failed(next_num_members);
194  std::vector<std::tuple<ip_addr_t, uint16_t, uint16_t, uint16_t, uint16_t>> member_ips_and_ports(next_num_members);
195  int next_unassigned_rank = curr_view.next_unassigned_rank;
196  for(std::size_t i = 0; i < joiner_ids.size(); ++i) {
197  joined.emplace_back(joiner_ids[i]);
198  //New members go at the end of the members list, but it may shrink in the new view
199  int new_member_rank = curr_view.num_members - leave_ranks.size() + i;
200  members[new_member_rank] = joiner_ids[i];
201  member_ips_and_ports[new_member_rank] = joiner_ips_and_ports[i];
202  }
203  for(const auto& leaver_rank : leave_ranks) {
204  departed.emplace_back(curr_view.members[leaver_rank]);
205  //Decrement next_unassigned_rank for every failure, unless the failure wasn't in a subgroup anyway
206  if(leaver_rank <= curr_view.next_unassigned_rank) {
207  next_unassigned_rank--;
208  }
209  }
210  //Copy member information, excluding the members that have failed
211  int m = 0;
212  for(int n = 0; n < curr_view.num_members; n++) {
213  //This is why leave_ranks needs to be a set
214  if(leave_ranks.find(n) == leave_ranks.end()) {
215  members[m] = curr_view.members[n];
216  member_ips_and_ports[m] = curr_view.member_ips_and_ports[n];
217  failed[m] = curr_view.failed[n];
218  ++m;
219  }
220  }
221 
222  //Initialize my_rank in next_view
223  int32_t my_new_rank = -1;
224  node_id_t myID = curr_view.members[curr_view.my_rank];
225  for(int i = 0; i < next_num_members; ++i) {
226  if(members[i] == myID) {
227  my_new_rank = i;
228  break;
229  }
230  }
231  return std::make_unique<View>(curr_view.vid + 1, members, member_ips_and_ports, failed,
232  joined, departed, my_new_rank, next_unassigned_rank,
233  curr_view.subgroup_type_order);
234 }
