168  const std::unique_ptr<View>& prev_view,
169  View& curr_view) {
170  int32_t initial_next_unassigned_rank = curr_view.next_unassigned_rank;
171  curr_view.subgroup_shard_views.clear();
172  curr_view.subgroup_ids_by_type_id.clear();
173  std::cout << "View has these members: " << curr_view.members << std::endl;
174  std::map<std::type_index, subgroup_shard_layout_t> subgroup_allocations;
175  try {
176  auto temp = subgroup_info.subgroup_membership_function(curr_view.subgroup_type_order,
177  prev_view, curr_view);
178  //Hack to ensure RVO works even though subgroup_allocations had to be declared outside this scope
179  subgroup_allocations = std::move(temp);
180  } catch(subgroup_provisioning_exception& ex) {
181  // Mark the view as inadequate and roll back everything done by allocation functions
182  curr_view.is_adequately_provisioned = false;
183  curr_view.next_unassigned_rank = initial_next_unassigned_rank;
184  curr_view.subgroup_shard_views.clear();
185  curr_view.subgroup_ids_by_type_id.clear();
186  std::cout << "Got a subgroup_provisioning_exception, marking View inadequate" << std::endl
187  << std::endl;
188  return;
189  }
190  //Go through subgroup_allocations and initialize curr_view
191  for(subgroup_type_id_t subgroup_type_id = 0;
192  subgroup_type_id < curr_view.subgroup_type_order.size();
193  ++subgroup_type_id) {
194  const std::type_index& subgroup_type = curr_view.subgroup_type_order[subgroup_type_id];
195  subgroup_shard_layout_t& curr_type_subviews = subgroup_allocations[subgroup_type];
196  std::cout << "Subgroup type " << subgroup_type.name() << " got assignment: " << std::endl;
197  print_subgroup_layout(curr_type_subviews);
198  std::size_t num_subgroups = curr_type_subviews.size();
199  curr_view.subgroup_ids_by_type_id.emplace(subgroup_type_id, std::vector<subgroup_id_t>(num_subgroups));
200  for(uint32_t subgroup_index = 0; subgroup_index < num_subgroups; ++subgroup_index) {
201  // Assign this (type, index) pair a new unique subgroup ID
202  subgroup_id_t curr_subgroup_id = curr_view.subgroup_shard_views.size();
203  curr_view.subgroup_ids_by_type_id[subgroup_type_id][subgroup_index] = curr_subgroup_id;
204  uint32_t num_shards = curr_type_subviews[subgroup_index].size();
205  for(uint shard_num = 0; shard_num < num_shards; ++shard_num) {
206  SubView& shard_view = curr_type_subviews[subgroup_index][shard_num];
207  shard_view.my_rank = shard_view.rank_of(curr_view.members[curr_view.my_rank]);
208  if(shard_view.my_rank != -1) {
209  // Initialize my_subgroups
210  curr_view.my_subgroups[curr_subgroup_id] = shard_num;
211  }
212  if(prev_view) {
213  // Initialize this shard's SubView.joined and SubView.departed
214  subgroup_id_t prev_subgroup_id = prev_view->subgroup_ids_by_type_id.at(subgroup_type_id)
215  .at(subgroup_index);
216  SubView& prev_shard_view = prev_view->subgroup_shard_views[prev_subgroup_id][shard_num];
217  shard_view.init_joined_departed(prev_shard_view);
218  }
219  } // for(shard_num)
220  /* Pull this shard->SubView mapping out of the subgroup allocation
221  * and save it under its subgroup ID (which was subgroup_shard_views.size()).
222  * This deletes it from the subgroup_shard_layout_t's outer vector. */
223  curr_view.subgroup_shard_views.emplace_back(std::move(
224  subgroup_allocations[subgroup_type][subgroup_index]));
225  } //for(subgroup_index)
226  }

229 std::unique_ptr<View> make_next_view(const View& curr_view,
230  const std::set<int>& leave_ranks,
231  const std::vector<node_id_t>& joiner_ids,
232  const std::vector<std::tuple<ip_addr_t, uint16_t, uint16_t, uint16_t, uint16_t>>& joiner_ips_and_ports) {
233  int next_num_members = curr_view.num_members - leave_ranks.size() + joiner_ids.size();
234  std::vector<node_id_t> joined, members(next_num_members), departed;
235  std::vector<char> failed(next_num_members);
236  std::vector<std::tuple<ip_addr_t, uint16_t, uint16_t, uint16_t, uint16_t>> member_ips_and_ports(next_num_members);
237  int next_unassigned_rank = curr_view.next_unassigned_rank;
238  for(std::size_t i = 0; i < joiner_ids.size(); ++i) {
239  joined.emplace_back(joiner_ids[i]);
240  //New members go at the end of the members list, but it may shrink in the new view
241  int new_member_rank = curr_view.num_members - leave_ranks.size() + i;
242  members[new_member_rank] = joiner_ids[i];
243  member_ips_and_ports[new_member_rank] = joiner_ips_and_ports[i];
244  }
245  for(const auto& leaver_rank : leave_ranks) {
246  departed.emplace_back(curr_view.members[leaver_rank]);
247  //Decrement next_unassigned_rank for every failure, unless the failure wasn't in a subgroup anyway
248  if(leaver_rank <= curr_view.next_unassigned_rank) {
249  next_unassigned_rank--;
250  }
251  }
252  //Copy member information, excluding the members that have failed
253  int m = 0;
254  for(int n = 0; n < curr_view.num_members; n++) {
255  //This is why leave_ranks needs to be a set
256  if(leave_ranks.find(n) == leave_ranks.end()) {
257  members[m] = curr_view.members[n];
258  member_ips_and_ports[m] = curr_view.member_ips_and_ports[n];
259  failed[m] = curr_view.failed[n];
260  ++m;
261  }
262  }
263 
264  //Initialize my_rank in next_view
265  int32_t my_new_rank = -1;
266  node_id_t myID = curr_view.members[curr_view.my_rank];
267  for(int i = 0; i < next_num_members; ++i) {
268  if(members[i] == myID) {
269  my_new_rank = i;
270  break;
271  }
272  }
273  return std::make_unique<View>(curr_view.vid + 1, members, member_ips_and_ports, failed,
274  joined, departed, my_new_rank, next_unassigned_rank,
275  curr_view.subgroup_type_order);
276 }
