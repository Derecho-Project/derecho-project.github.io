292  // These fields had better be synchronized.
293  assert(gmsSST.get_local_index() == curr_view->my_rank);
294  // Aggregate suspicions into gmsSST[myRank].Suspected;
295  for(int r = 0; r < Vc.num_members; r++) {
296  for(int who = 0; who < Vc.num_members; who++) {
297  gmssst::set(gmsSST.suspected[myRank][who], gmsSST.suspected[myRank][who] || gmsSST.suspected[r][who]);
298  }
299  }
300 
301  for(int q = 0; q < Vc.num_members; q++) {
302  //If this is a new suspicion
303  if(gmsSST.suspected[myRank][q] && !Vc.failed[q]) {
304  //This is safer than copy_suspected, since suspected[] might change during this loop
305  last_suspected[q] = gmsSST.suspected[myRank][q];
306  logger->debug("Marking {} failed", Vc.members[q]);
307  if(Vc.num_failed >= (Vc.num_members + 1) / 2) {
308  throw derecho_exception("Majority of a Derecho group simultaneously failed ... shutting down");
309  }
310 
311  logger->debug("GMS telling SST to freeze row {} which is node {}", q, Vc.members[q]);
312  gmsSST.freeze(q); // Cease to accept new updates from q
313  Vc.multicast_group->wedge();
314  gmssst::set(gmsSST.wedged[myRank], true); // RDMC has halted new sends and receives in theView
315  Vc.failed[q] = true;
316  Vc.num_failed++;
317 
318  if(Vc.num_failed >= (Vc.num_members + 1) / 2) {
319  throw derecho_exception("Potential partitioning event: this node is no longer in the majority and must shut down!");

322  // push change to gmsSST.suspected[myRank]

325  gmsSST.put(gmsSST.wedged.get_base() - gmsSST.getBaseAddress(), sizeof(bool));

334  gmssst::increment(gmsSST.num_changes[myRank]);

336  gmsSST.put((char*)std::addressof(gmsSST.changes[0][next_change_index]) - gmsSST.getBaseAddress(),

490  next_view->i_know_i_am_leader = Vc.i_know_i_am_leader;
491 
492  // At this point we need to await "meta wedged."
493  // To do that, we create a predicate that will fire when meta wedged is
494  // true, and put the rest of the code in its trigger.
495 
496  auto is_meta_wedged = [this](const DerechoSST& gmsSST) {
497  for(int n = 0; n < gmsSST.get_num_rows(); ++n) {
498  if(!curr_view->failed[n] && !gmsSST.wedged[n]) {
499  return false;
500  }
501  }
502  return true;
503  };
504  auto meta_wedged_continuation = [this](DerechoSST& gmsSST) {
505  logger->debug("MetaWedged is true; continuing view change");
506  std::unique_lock<std::shared_timed_mutex> write_lock(view_mutex);
507  assert(next_view);
508 
509  //First, for subgroups in which I'm the shard leader, do RaggedEdgeCleanup for the leader
510  auto follower_subgroups_and_shards = std::make_shared<std::map<subgroup_id_t, uint32_t>>();
511  for(const auto& shard_rank_pair : curr_view->multicast_group->get_subgroup_to_shard_and_rank()) {
512  const subgroup_id_t subgroup_id = shard_rank_pair.first;
513  const uint32_t shard_num = shard_rank_pair.second.first;
514  SubView& shard_view = *curr_view->subgroup_shard_views.at(subgroup_id).at(shard_num);
515  uint num_shard_senders = 0;
516  for(auto v : shard_view.is_sender) {
517  if(v) num_shard_senders++;
518  }
519  if(num_shard_senders) {
520  if(shard_view.my_rank == curr_view->subview_rank_of_shard_leader(subgroup_id, shard_num)) {
521  leader_ragged_edge_cleanup(*curr_view, subgroup_id,
522  curr_view->multicast_group->get_subgroup_to_num_received_offset()
523  .at(subgroup_id),
524  shard_view.members, num_shard_senders);
525  } else {
526  //Keep track of which subgroups I'm a non-leader in, and what my corresponding shard ID is
527  follower_subgroups_and_shards->emplace(subgroup_id, shard_num);
528  }
529  }
530  }
531 
532  //Wait for the shard leaders of subgroups I'm not a leader in to post global_min_ready before continuing
533  auto leader_global_mins_are_ready = [this, follower_subgroups_and_shards](const DerechoSST& gmsSST) {
534  for(const auto& subgroup_shard_pair : *follower_subgroups_and_shards) {
535  SubView& shard_view = *curr_view->subgroup_shard_views.at(subgroup_shard_pair.first)
536  .at(subgroup_shard_pair.second);
537  node_id_t shard_leader = shard_view.members[curr_view->subview_rank_of_shard_leader(
538  subgroup_shard_pair.first, subgroup_shard_pair.second)];
539  if(!gmsSST.global_min_ready[curr_view->rank_of(shard_leader)][subgroup_shard_pair.first])
540  return false;
541  }
542  return true;
543  };
544 
545  auto global_min_ready_continuation = [this, follower_subgroups_and_shards](DerechoSST& gmsSST) {
546  std::unique_lock<std::shared_timed_mutex> write_lock(view_mutex);
547  assert(next_view);
548 
549  logger->debug("GlobalMins are ready for all {} subgroup leaders this node is waiting on", follower_subgroups_and_shards->size());
550  //Finish RaggedEdgeCleanup for subgroups in which I'm not the leader
551  for(const auto& subgroup_shard_pair : *follower_subgroups_and_shards) {
552  SubView& shard_view = *curr_view->subgroup_shard_views.at(subgroup_shard_pair.first)
553  .at(subgroup_shard_pair.second);
554  uint num_shard_senders = 0;
555  for(auto v : shard_view.is_sender) {
556  if(v) num_shard_senders++;
557  }
558  node_id_t shard_leader = shard_view.members[curr_view->subview_rank_of_shard_leader(
559  subgroup_shard_pair.first, subgroup_shard_pair.second)];
560  follower_ragged_edge_cleanup(*curr_view, subgroup_shard_pair.first,
561  curr_view->rank_of(shard_leader),
562  curr_view->multicast_group->get_subgroup_to_num_received_offset()
563  .at(subgroup_shard_pair.first),
564  shard_view.members,
565  num_shard_senders);
566  }
567  //Calculate and save the IDs of shard leaders for the old view
568  //If the old view was inadequately provisioned, this will be empty
569  std::map<std::type_index, std::vector<std::vector<int64_t>>> old_shard_leaders_by_type
570  = make_shard_leaders_map(*curr_view);
571

573  if(curr_view->i_am_leader() && next_view->joined.size() > 0) {
574  //If j joins have been committed, pop the next j sockets off proposed_join_sockets
575  //and send them the new View (must happen before we try to do SST setup)
576  for(std::size_t c = 0; c < next_view->joined.size(); ++c) {
577  commit_join(*next_view, proposed_join_sockets.front());

579  joiner_sockets.emplace_back(std::move(proposed_join_sockets.front()));
580  proposed_join_sockets.pop_front();
581  }
582  }
583

585  gmsSST.predicates.remove(leader_committed_handle);
586  gmsSST.predicates.remove(suspected_changed_handle);
587 
588  node_id_t my_id = next_view->members[next_view->my_rank];
589  logger->debug("Starting creation of new SST and DerechoGroup for view {}", next_view->vid);
590  // if new members have joined, add their RDMA connections to SST and RDMC
591  for(std::size_t i = 0; i < next_view->joined.size(); ++i) {
592  //The new members will be the last joined.size() elements of the members lists
593  int joiner_rank = next_view->num_members - next_view->joined.size() + i;
594  rdma::impl::verbs_add_connection(next_view->members[joiner_rank], next_view->member_ips[joiner_rank],
595  my_id);
596  }
597  for(std::size_t i = 0; i < next_view->joined.size(); ++i) {
598  int joiner_rank = next_view->num_members - next_view->joined.size() + i;
599  sst::add_node(next_view->members[joiner_rank], next_view->member_ips[joiner_rank]);
600  }
601  // This will block until everyone responds to SST/RDMC initial handshakes
602  transition_multicast_group();
603 
604  // Translate the old shard leaders' indices from types to new subgroup IDs
605  std::vector<std::vector<int64_t>> old_shard_leaders_by_id = translate_types_to_ids(old_shard_leaders_by_type, *next_view);
606 
607  if(curr_view->i_am_leader()) {
608  while(!joiner_sockets.empty()) {
609  //Send the array of old shard leaders, so the new member knows who to receive from
610  std::size_t size_of_vector = mutils::bytes_size(old_shard_leaders_by_id);
611  joiner_sockets.front().write((char*)&size_of_vector, sizeof(std::size_t));
612  mutils::post_object([&joiner_sockets](const char* bytes, std::size_t size) {
613  joiner_sockets.front().write(bytes, size);
614  },
615  old_shard_leaders_by_id);
616  joiner_sockets.pop_front();
617  }
618  }
619 
620  // New members can now proceed to view_manager.start(), which will call sync()
621  next_view->gmsSST->put();
622  next_view->gmsSST->sync_with_members();
623  logger->debug("Done setting up SST and DerechoGroup for view {}", next_view->vid);
624  {
625  lock_guard_t old_views_lock(old_views_mutex);
626  old_views.push(std::move(curr_view));
627  old_views_cv.notify_all();
628  }
629  curr_view = std::move(next_view);
630 
631  //If in persistent mode, write the new view to disk before using it
632  if(!view_file_name.empty()) {
633  persist_object(*curr_view, view_file_name);
634  }
635 
636  //Resize last_suspected to match the new size of suspected[]
637  last_suspected.resize(curr_view->members.size());
638 
639  // Register predicates in the new view
640  register_predicates();
641  curr_view->gmsSST->start_predicate_evaluation();
642 
643  // First task with my new view...
644  if(curr_view->i_am_new_leader()) // I'm the new leader and everyone who hasn't failed agrees
645  {
646  curr_view->merge_changes(); // Create a combined list of Changes
647  }
648 
649  // Announce the new view to the application
650  for(auto& view_upcall : view_upcalls) {
651  view_upcall(*curr_view);
652  }
653  // One of those view upcalls is to RPCManager, which will set up TCP connections to the new members
654  // After doing that, shard leaders can send them RPC objects
655  for(subgroup_id_t subgroup_id = 0; subgroup_id < old_shard_leaders_by_id.size(); ++subgroup_id) {
656  for(uint32_t shard = 0; shard < old_shard_leaders_by_id[subgroup_id].size(); ++shard) {
657  //if I was the leader of the shard in the old view...
658  if(my_id == old_shard_leaders_by_id[subgroup_id][shard]) {
659  //send its object state to the new members
660  for(node_id_t shard_joiner : curr_view->subgroup_shard_views[subgroup_id][shard]->joined) {
661  if(shard_joiner != my_id) {
662  send_subgroup_object(subgroup_id, shard_joiner);
663  }
664  }
665  }
666  }
667  }
668 
669  // Re-initialize this node's RPC objects, which includes receiving them
670  // from shard leaders if it is newly a member of a subgroup
671  initialize_subgroup_objects(my_id, *curr_view, old_shard_leaders_by_id);
672  view_change_cv.notify_all();
673  };
674 
675  //Last statement in meta_wedged_continuation: register global_min_ready_continuation
676  gmsSST.predicates.insert(leader_global_mins_are_ready, global_min_ready_continuation, sst::PredicateType::ONE_TIME);
677 
678  };
679 
680  //Last statement in start_view_change: register meta_wedged_continuation
681  gmsSST.predicates.insert(is_meta_wedged, meta_wedged_continuation, sst::PredicateType::ONE_TIME);
682 
683  };
684 
685  suspected_changed_handle = curr_view->gmsSST->predicates.insert(suspected_changed, suspected_changed_trig, sst::PredicateType::RECURRENT);
686  start_join_handle = curr_view->gmsSST->predicates.insert(start_join_pred, start_join_trig, sst::PredicateType::RECURRENT);
687  change_commit_ready_handle = curr_view->gmsSST->predicates.insert(change_commit_ready, commit_change, sst::PredicateType::RECURRENT);
688  leader_proposed_handle = curr_view->gmsSST->predicates.insert(leader_proposed_change, ack_proposed_change, sst::PredicateType::RECURRENT);
689  leader_committed_handle = curr_view->gmsSST->predicates.insert(leader_committed_next_view, start_view_change, sst::PredicateType::ONE_TIME);
690 }

768  curr_view->wedge();

784 uint32_t ViewManager::make_subgroup_maps(const std::unique_ptr<View>& prev_view,
785  View& curr_view,
786  std::map<subgroup_id_t, std::pair<uint32_t, uint32_t>>& subgroup_to_shard_and_rank,
787  std::map<subgroup_id_t, std::pair<std::vector<int>, int>>& subgroup_to_senders_and_sender_rank,
788  std::map<subgroup_id_t, uint32_t>& subgroup_to_num_received_offset,
789  std::map<subgroup_id_t, std::vector<node_id_t>>& subgroup_to_membership) {
790  uint32_t subgroup_offset = 0;
791  for(const auto& subgroup_type_and_function : subgroup_info.subgroup_membership_functions) {
792  subgroup_shard_layout_t subgroup_shard_views;
793  //This is the only place the subgroup membership functions are called; the results are then saved in the View
794  try {
795  auto temp = subgroup_type_and_function.second(curr_view);
796  //Hack to ensure RVO still works even though subgroup_shard_views had to be declared outside this scope
797  subgroup_shard_views = std::move(temp);
798  } catch(subgroup_provisioning_exception& ex) {
799  curr_view.is_adequately_provisioned = false;
800  subgroup_to_shard_and_rank.clear();
801  subgroup_to_num_received_offset.clear();
802  subgroup_to_membership.clear();
803  curr_view.subgroup_shard_views.clear();
804  curr_view.subgroup_ids_by_type.clear();
805  return 0;
806  }
807  std::size_t num_subgroups = subgroup_shard_views.size();
808  curr_view.subgroup_ids_by_type[subgroup_type_and_function.first] = std::vector<subgroup_id_t>(num_subgroups);
809  for(uint32_t subgroup_index = 0; subgroup_index < num_subgroups; ++subgroup_index) {
810  //Assign this (type, index) pair a new unique subgroup ID
811  subgroup_id_t next_subgroup_number = curr_view.subgroup_shard_views.size();
812  curr_view.subgroup_ids_by_type[subgroup_type_and_function.first][subgroup_index] = next_subgroup_number;
813  uint32_t num_shards = subgroup_shard_views.at(subgroup_index).size();
814  uint32_t max_shard_senders = 0;
815  for(uint shard_num = 0; shard_num < num_shards; ++shard_num) {
816  SubView& shard_view = *subgroup_shard_views.at(subgroup_index).at(shard_num);
817  std::size_t shard_size = shard_view.members.size();
818  uint32_t num_shard_senders = shard_view.num_senders();
819  if(num_shard_senders > max_shard_senders) {
820  max_shard_senders = shard_size;
821  }
822  //Initialize my_rank in the SubView for this node's ID
823  shard_view.my_rank = shard_view.rank_of(curr_view.members[curr_view.my_rank]);
824  if(shard_view.my_rank != -1) {
825  subgroup_to_shard_and_rank[next_subgroup_number] = {shard_num, shard_view.my_rank};
826  subgroup_to_senders_and_sender_rank[next_subgroup_number] = {shard_view.is_sender, shard_view.sender_rank_of(shard_view.my_rank)};
827  subgroup_to_num_received_offset[next_subgroup_number] = subgroup_offset;
828  subgroup_to_membership[next_subgroup_number] = shard_view.members;
829  }
830  if(prev_view && prev_view->is_adequately_provisioned) {
831  //Initialize this shard's SubView.joined and SubView.departed
832  subgroup_id_t prev_subgroup_id = prev_view->subgroup_ids_by_type
833  .at(subgroup_type_and_function.first)
834  .at(subgroup_index);
835  SubView& prev_shard_view = *prev_view->subgroup_shard_views[prev_subgroup_id][shard_num];
836  std::set<node_id_t> prev_members(prev_shard_view.members.begin(), prev_shard_view.members.end());
837  std::set<node_id_t> curr_members(shard_view.members.begin(), shard_view.members.end());
838  std::set_difference(curr_members.begin(), curr_members.end(),
839  prev_members.begin(), prev_members.end(),
840  std::back_inserter(shard_view.joined));
841  std::set_difference(prev_members.begin(), prev_members.end(),
842  curr_members.begin(), curr_members.end(),
843  std::back_inserter(shard_view.departed));
844  }
845  }
846  /* Pull the shard->SubView mapping out of the subgroup membership list
847  * and save it under its subgroup ID (which was shard_views_by_subgroup.size()) */
848  curr_view.subgroup_shard_views.emplace_back(
849  std::move(subgroup_shard_views[subgroup_index]));
850  subgroup_offset += max_shard_senders;
851  }
852  }
853  return subgroup_offset;
854 }

975  const uint32_t num_received_offset,
976  const std::vector<node_id_t>& shard_members, uint num_shard_senders) {
977  logger->debug("Running leader RaggedEdgeCleanup for subgroup {}", subgroup_num);
978  int myRank = Vc.my_rank;
979  // int Leader = Vc.rank_of_leader(); // We don't want this to change under our feet
980  bool found = false;
981  for(uint n = 0; n < shard_members.size() && !found; n++) {
982  const auto node_id = shard_members[n];
983  const auto node_rank = Vc.rank_of(node_id);
984  if(Vc.gmsSST->global_min_ready[node_rank][subgroup_num]) {
985  gmssst::set(Vc.gmsSST->global_min[myRank] + num_received_offset,
986  Vc.gmsSST->global_min[node_rank] + num_received_offset, num_shard_senders);
987  found = true;
988  }
989  }
990 
991  if(!found) {
992  for(uint n = 0; n < num_shard_senders; n++) {
993  int min = Vc.gmsSST->num_received[myRank][num_received_offset + n];
994  for(uint r = 0; r < shard_members.size(); r++) {

998  min = Vc.gmsSST->num_received[node_rank][num_received_offset + n];
999  }
1000  }
1001 
1002  gmssst::set(Vc.gmsSST->global_min[myRank][num_received_offset + n], min);
1003  }
1004  }
1005

1007  gmssst::set(Vc.gmsSST->global_min_ready[myRank][subgroup_num], true);

1009  (char*)std::addressof(Vc.gmsSST->global_min[0][num_received_offset]) - Vc.gmsSST->getBaseAddress(),
1010  sizeof(int) * num_shard_senders);

1013  sizeof(bool));
1014
