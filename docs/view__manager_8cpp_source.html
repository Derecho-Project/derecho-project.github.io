140 bool ViewManager::receive_configuration(node_id_t my_id, tcp::socket& leader_connection) {
141  JoinResponse leader_response;
142  bool leader_redirect;
143  do {
144  leader_redirect = false;
145  bool success;
146  whenlog(logger->debug("Socket connected to leader, exchanging IDs."););
147  success = leader_connection.write(my_id);
148  if(!success) throw derecho_exception("Failed to exchange IDs with the leader! Leader has crashed.");
149  success = leader_connection.read(leader_response);
150  if(!success) throw derecho_exception("Failed to exchange IDs with the leader! Leader has crashed.");
151  if(leader_response.code == JoinResponseCode::ID_IN_USE) {
152  whenlog(logger->error("Error! Leader refused connection because ID {} is already in use!", my_id););
153  whenlog(logger->flush();) throw derecho_exception("Leader rejected join, ID already in use");
154  }
155  if(leader_response.code == JoinResponseCode::LEADER_REDIRECT) {
156  std::size_t ip_addr_size;
157  leader_connection.read(ip_addr_size);
158  char buffer[ip_addr_size];
159  leader_connection.read(buffer, ip_addr_size);
160  ip_addr_t leader_ip(buffer);
161  uint16_t leader_gms_port;
162  leader_connection.read(leader_gms_port);
163  whenlog(logger->info("That node was not the leader! Redirecting to {}", leader_ip););
164  //Use move-assignment to reconnect the socket to the given IP address, and try again
165  //(good thing that leader_connection reference is mutable)
166  leader_connection = tcp::socket(leader_ip, leader_gms_port);
167  leader_redirect = true;
168  }
169  } while(leader_redirect);
170 
171  bool is_total_restart = (leader_response.code == JoinResponseCode::TOTAL_RESTART);
172  if(is_total_restart) {
173  whenlog(logger->debug("In restart mode, sending view {} to leader", curr_view->vid););
174  bool success = leader_connection.write(mutils::bytes_size(*curr_view));
175  if(!success) throw derecho_exception("Restart leader crashed before sending a restart View!");
176  auto leader_socket_write = [&leader_connection](const char* bytes, std::size_t size) {
177  if(!leader_connection.write(bytes, size)) {
178  throw derecho_exception("Restart leader crashed before sending a restart View!");
179  }
180  };
181  mutils::post_object(leader_socket_write, *curr_view);
182  //Restore this non-serializeable field to curr_view before using it
183  curr_view->subgroup_type_order = subgroup_type_order;
184  restart_state = std::make_unique<RestartState>();
185  restart_state->load_ragged_trim(*curr_view);
186  whenlog(logger->debug("In restart mode, sending {} ragged trims to leader", restart_state->logged_ragged_trim.size()););
187  /* Protocol: Send the number of RaggedTrim objects, then serialize each RaggedTrim */
188  /* Since we know this node is only a member of one shard per subgroup,
189  * the size of the outer map (subgroup IDs) is the number of RaggedTrims. */
190  success
191  = leader_connection.write(restart_state->logged_ragged_trim.size());
192  if(!success) throw derecho_exception("Restart leader crashed before sending a restart View!");
193  for(const auto& id_to_shard_map : restart_state->logged_ragged_trim) {
194  const std::unique_ptr<RaggedTrim>& ragged_trim = id_to_shard_map.second.begin()->second; //The inner map has one entry
195  success = leader_connection.write(mutils::bytes_size(*ragged_trim));
196  if(!success) throw derecho_exception("Restart leader crashed before sending a restart View!");
197  mutils::post_object(leader_socket_write, *ragged_trim);
198  }
199  }
200  leader_connection.write(getConfUInt16(CONF_DERECHO_GMS_PORT));
201  leader_connection.write(getConfUInt16(CONF_DERECHO_RPC_PORT));
202  leader_connection.write(getConfUInt16(CONF_DERECHO_SST_PORT));
203  leader_connection.write(getConfUInt16(CONF_DERECHO_RDMC_PORT));
204 
205  bool view_confirmed = false;
206  while(!view_confirmed) {
207  //The leader will first send the size of the necessary buffer, then the serialized View
208  std::size_t size_of_view;
209  bool success = leader_connection.read(size_of_view);
210  if(!success) {
211  throw derecho_exception("Leader crashed before it could send the initial View! Try joining again at the new leader.");
212  }
213  char buffer[size_of_view];
214  success = leader_connection.read(buffer, size_of_view);
215  if(!success) {
216  throw derecho_exception("Leader crashed before it could send the initial View! Try joining again at the new leader.");
217  }
218  curr_view = mutils::from_bytes<View>(nullptr, buffer);
219  //Next, the leader sends DerechoParams
220  std::size_t size_of_derecho_params;
221  success = leader_connection.read(size_of_derecho_params);
222  char buffer2[size_of_derecho_params];
223  success = leader_connection.read(buffer2, size_of_derecho_params);
224  if(!success) {
225  throw derecho_exception("Leader crashed before it could send the initial View! Try joining again at the new leader.");
226  }
227  derecho_params = *mutils::from_bytes<DerechoParams>(nullptr, buffer2);
228  if(is_total_restart) {
229  //In total restart mode, the leader will also send the RaggedTrims it has collected
230  whenlog(logger->debug("In restart mode, receiving ragged trim from leader"););
231  restart_state->logged_ragged_trim.clear();
232  std::size_t num_of_ragged_trims;
233  leader_connection.read(num_of_ragged_trims);
234  for(std::size_t i = 0; i < num_of_ragged_trims; ++i) {
235  std::size_t size_of_ragged_trim;
236  leader_connection.read(size_of_ragged_trim);
237  char buffer[size_of_ragged_trim];
238  leader_connection.read(buffer, size_of_ragged_trim);
239  std::unique_ptr<RaggedTrim> ragged_trim = mutils::from_bytes<RaggedTrim>(nullptr, buffer);
240  //operator[] is intentional: Create an empty inner map at subgroup_id if one does not exist
241  restart_state->logged_ragged_trim[ragged_trim->subgroup_id].emplace(ragged_trim->shard_num, std::move(ragged_trim));
242  }
243  }
244  //Next, the leader will send a single bool indicating whether this View has been committed
245  //at all newly joining members. If not, we must go back to waiting for a View.
246  success = leader_connection.read(view_confirmed);
247  if(!success) {
248  throw derecho_exception("Leader crashed before it could send the initial View! Try joining again at the new leader.");
249  }
250  whenlog(logger->debug("Received view {} from leader. View_confirmed = {}", curr_view->vid, view_confirmed););
251  }
252  //This must be set up locally, since it's not serializable
253  curr_view->subgroup_type_order = subgroup_type_order;
254  return is_total_restart;
255 }

395  std::vector<char>(curr_view->num_members + 1, 0),
396  functional_append(curr_view->joined, joiner_id),
397  std::vector<node_id_t>{}, 0, 0,
398  subgroup_type_order);

401  whenlog(logger->debug("Node {} connected from IP address {} and GMS port {}", joiner_id, joiner_ip, joiner_gms_port););
402  }
403  joiner_failed = false;

428  if(!send_success) {

430  }

432  waiting_sockets_iter++;
433  } catch(node_id_t failed_joiner_id) {
434  whenlog(logger->info("Node {} failed after contacting the leader! Removing it from the initial view.", failed_joiner_id););
435  //Remove the failed client and recompute the view
436  std::vector<node_id_t> filtered_members(curr_view->members.size() - 1);
437  std::vector<std::tuple<ip_addr_t, uint16_t, uint16_t, uint16_t, uint16_t>> filtered_ips_and_ports(curr_view->member_ips_and_ports.size() - 1);
438  std::vector<node_id_t> filtered_joiners(curr_view->joined.size() - 1);
439  std::remove_copy(curr_view->members.begin(), curr_view->members.end(),
440  filtered_members.begin(), failed_joiner_id);
441  std::remove_copy(curr_view->member_ips_and_ports.begin(), curr_view->member_ips_and_ports.end(),
442  filtered_ips_and_ports.begin(), curr_view->member_ips_and_ports[curr_view->rank_of(failed_joiner_id)]);

447  std::vector<node_id_t>{}, 0, 0,

455  }

468  waiting_sockets_iter != waiting_join_sockets.end();) {

471  }

573 void ViewManager::register_predicates() {
574  /* Note that each trigger function must be wrapped in a lambda because it's
575  * a member function, and lambdas are the only way to bind "this" to a member
576  * function invocation. */
577  auto suspected_changed = [this](const DerechoSST& sst) {
578  return suspected_not_equal(sst, last_suspected);
579  };
580  auto suspected_changed_trig = [this](DerechoSST& sst) { new_suspicion(sst); };
581 
582  auto start_join_pred = [this](const DerechoSST& sst) {
583  return curr_view->i_am_leader() && has_pending_join();
584  };
585  auto start_join_trig = [this](DerechoSST& sst) { leader_start_join(sst); };
586 
587  auto reject_join_pred = [this](const DerechoSST& sst) {
588  return !curr_view->i_am_leader() && has_pending_join();
589  };
590  auto reject_join = [this](DerechoSST& sst) { redirect_join_attempt(sst); };
591 
592  auto change_commit_ready = [this](const DerechoSST& gmsSST) {
593  return curr_view->i_am_leader()
594  && min_acked(gmsSST, curr_view->failed) > gmsSST.num_committed[gmsSST.get_local_index()];
595  };
596  auto commit_change = [this](DerechoSST& sst) { leader_commit_change(sst); };
597 
598  auto leader_proposed_change = [this](const DerechoSST& gmsSST) {
599  return gmsSST.num_changes[curr_view->rank_of_leader()]
600  > gmsSST.num_acked[gmsSST.get_local_index()];
601  };
602  auto ack_proposed_change = [this](DerechoSST& sst) {
603  acknowledge_proposed_change(sst);
604  };
605 
606  auto leader_committed_changes = [this](const DerechoSST& gmsSST) {
607  return gmsSST.num_committed[curr_view->rank_of_leader()]
608  > gmsSST.num_installed[curr_view->my_rank];
609  };
610  auto view_change_trig = [this](DerechoSST& sst) { start_meta_wedge(sst); };
611 
612  if(!suspected_changed_handle.is_valid()) {
613  suspected_changed_handle = curr_view->gmsSST->predicates.insert(
614  suspected_changed, suspected_changed_trig,
615  sst::PredicateType::RECURRENT);
616  }
617  if(!start_join_handle.is_valid()) {
618  start_join_handle = curr_view->gmsSST->predicates.insert(
619  start_join_pred, start_join_trig, sst::PredicateType::RECURRENT);
620  }
621  if(!reject_join_handle.is_valid()) {
622  reject_join_handle = curr_view->gmsSST->predicates.insert(reject_join_pred, reject_join,
623  sst::PredicateType::RECURRENT);
624  }
625  if(!change_commit_ready_handle.is_valid()) {
626  change_commit_ready_handle = curr_view->gmsSST->predicates.insert(
627  change_commit_ready, commit_change, sst::PredicateType::RECURRENT);
628  }
629  if(!leader_proposed_handle.is_valid()) {
630  leader_proposed_handle = curr_view->gmsSST->predicates.insert(
631  leader_proposed_change, ack_proposed_change,
632  sst::PredicateType::RECURRENT);
633  }
634  if(!leader_committed_handle.is_valid()) {
635  leader_committed_handle = curr_view->gmsSST->predicates.insert(
636  leader_committed_changes, view_change_trig,
637  sst::PredicateType::ONE_TIME);
638  }

643 void ViewManager::new_suspicion(DerechoSST& gmsSST) {
644  whenlog(logger->debug("Suspected[] changed"););
645  View& Vc = *curr_view;
646  int myRank = curr_view->my_rank;
647  // Aggregate suspicions into gmsSST[myRank].Suspected;
648  for(int r = 0; r < Vc.num_members; r++) {
649  for(int who = 0; who < Vc.num_members; who++) {
650  gmssst::set(gmsSST.suspected[myRank][who],
651  gmsSST.suspected[myRank][who] || gmsSST.suspected[r][who]);
652  }
653  }
654 
655  for(int q = 0; q < Vc.num_members; q++) {
656  // if(gmsSST.suspected[myRank][q] && !Vc.failed[q]) {
657  if(gmsSST.suspected[myRank][q] && !last_suspected[q]) {
658  // This is safer than copy_suspected, since suspected[] might change
659  // during this loop
660  last_suspected[q] = gmsSST.suspected[myRank][q];
661  whenlog(logger->debug("Marking {} failed", Vc.members[q]););
662 
663  if(Vc.num_failed >= (Vc.num_members + 1) / 2) {
664  throw derecho_exception("Majority of a Derecho group simultaneously failed ... shutting down");
665  }
666 
667  whenlog(logger->debug("GMS telling SST to freeze row {}", q););
668  gmsSST.freeze(q); // Cease to accept new updates from q
669  Vc.multicast_group->wedge();
670  gmssst::set(gmsSST.wedged[myRank],
671  true); // RDMC has halted new sends and receives in theView
672  Vc.failed[q] = true;
673  Vc.num_failed++;
674 
675  if(Vc.num_failed >= (Vc.num_members + 1) / 2) {
676  throw derecho_exception("Potential partitioning event: this node is no longer in the majority and must shut down!");
677  }
678 
679  // push change to gmsSST.suspected[myRank]
680  gmsSST.put(gmsSST.suspected.get_base() - gmsSST.getBaseAddress(),
681  gmsSST.changes.get_base() - gmsSST.suspected.get_base());
682  // push change to gmsSST.wedged[myRank]
683  gmsSST.put(gmsSST.wedged.get_base() - gmsSST.getBaseAddress(),
684  sizeof(gmsSST.wedged[0]));
685  if(Vc.i_am_leader() && !changes_contains(gmsSST, Vc.members[q])) // Leader initiated
686  {
687  const int next_change_index = gmsSST.num_changes[myRank] - gmsSST.num_installed[myRank];
688  if(next_change_index == (int)gmsSST.changes.size()) {
689  throw derecho_exception("Ran out of room in the pending changes list");
690  }
691 
692  gmssst::set(
693  gmsSST.changes[myRank][next_change_index],
694  Vc.members[q]); // Reports the failure (note that q NotIn members)
695  gmssst::increment(gmsSST.num_changes[myRank]);
696  whenlog(logger->debug("Leader proposed a change to remove failed node {}", Vc.members[q]););
697  gmsSST.put(
698  (char*)std::addressof(gmsSST.changes[0][next_change_index]) - gmsSST.getBaseAddress(),
699  sizeof(gmsSST.changes[0][next_change_index]));
700  gmsSST.put(gmsSST.num_changes.get_base() - gmsSST.getBaseAddress(),
701  sizeof(gmsSST.num_changes[0]));
702  }
703  }
704  }

744 void ViewManager::acknowledge_proposed_change(DerechoSST& gmsSST) {
745  int myRank = gmsSST.get_local_index();
746  int leader = curr_view->rank_of_leader();
747  whenlog(logger->debug("Detected that leader proposed change #{}. Acknowledging.", gmsSST.num_changes[leader]););
748  if(myRank != leader) {
749  // Echo the count
750  gmssst::set(gmsSST.num_changes[myRank], gmsSST.num_changes[leader]);
751 
752  // Echo (copy) the vector including the new changes
753  gmssst::set(gmsSST.changes[myRank], gmsSST.changes[leader],
754  gmsSST.changes.size());
755  // Echo the new member's IP and ports
756  gmssst::set(gmsSST.joiner_ips[myRank], gmsSST.joiner_ips[leader],
757  gmsSST.joiner_ips.size());
758  gmssst::set(gmsSST.joiner_gms_ports[myRank], gmsSST.joiner_gms_ports[leader],
759  gmsSST.joiner_gms_ports.size());
760  gmssst::set(gmsSST.joiner_rpc_ports[myRank], gmsSST.joiner_rpc_ports[leader],
761  gmsSST.joiner_rpc_ports.size());
762  gmssst::set(gmsSST.joiner_sst_ports[myRank], gmsSST.joiner_sst_ports[leader],
763  gmsSST.joiner_sst_ports.size());
764  gmssst::set(gmsSST.joiner_rdmc_ports[myRank], gmsSST.joiner_rdmc_ports[leader],
765  gmsSST.joiner_rdmc_ports.size());
766  gmssst::set(gmsSST.num_committed[myRank], gmsSST.num_committed[leader]);
767  }
768 
769  // Notice a new request, acknowledge it
770  gmssst::set(gmsSST.num_acked[myRank], gmsSST.num_changes[myRank]);
771  /* breaking the above put statement into individual put calls, to be sure that
772  * if we were relying on any ordering guarantees, we won't run into issue when
773  * guarantees do not hold*/
774  gmsSST.put(gmsSST.changes.get_base() - gmsSST.getBaseAddress(),
775  gmsSST.joiner_ips.get_base() - gmsSST.changes.get_base());
776  gmsSST.put(gmsSST.joiner_ips.get_base() - gmsSST.getBaseAddress(),
777  gmsSST.num_changes.get_base() - gmsSST.joiner_ips.get_base());
778  gmsSST.put(gmsSST.num_changes.get_base() - gmsSST.getBaseAddress(),
779  gmsSST.num_committed.get_base() - gmsSST.num_changes.get_base());
780  gmsSST.put(gmsSST.num_committed.get_base() - gmsSST.getBaseAddress(),
781  gmsSST.num_acked.get_base() - gmsSST.num_committed.get_base());
782  gmsSST.put(gmsSST.num_acked.get_base() - gmsSST.getBaseAddress(),
783  gmsSST.num_installed.get_base() - gmsSST.num_acked.get_base());
784  gmsSST.put(gmsSST.num_installed.get_base() - gmsSST.getBaseAddress(),
785  gmsSST.num_received.get_base() - gmsSST.num_installed.get_base());
786  whenlog(logger->debug("Wedging current view."););
787  curr_view->wedge();
788  whenlog(logger->debug("Done wedging current view."););
789 }

791 void ViewManager::start_meta_wedge(DerechoSST& gmsSST) {
792  whenlog(logger->debug("Meta-wedging view {}", curr_view->vid););
793  // Disable all the other SST predicates, except suspected_changed and the
794  // one I'm about to register
795  gmsSST.predicates.remove(start_join_handle);
796  gmsSST.predicates.remove(reject_join_handle);
797  gmsSST.predicates.remove(change_commit_ready_handle);
798  gmsSST.predicates.remove(leader_proposed_handle);
799 
800  curr_view->wedge();
801 
802  /* We now need to wait for all other nodes to wedge the current view,
803  * which is called "meta-wedged." To do that, this predicate trigger
804  * creates a new predicate that will fire when meta-wedged is true, and
805  * registers the next epoch termination method as its trigger.
806  */
807  auto is_meta_wedged = [this](const DerechoSST& gmsSST) {
808  for(unsigned int n = 0; n < gmsSST.get_num_rows(); ++n) {
809  if(!curr_view->failed[n] && !gmsSST.wedged[n]) {
810  return false;
811  }
812  }
813  return true;
814  };
815  auto meta_wedged_continuation = [this](DerechoSST& gmsSST) {
816  // Before the first call to terminate_epoch(), heap-allocate this map
817  auto next_subgroup_settings = std::make_shared<std::map<subgroup_id_t, SubgroupSettings>>();
818  terminate_epoch(next_subgroup_settings, 0, gmsSST);
819  };
820  gmsSST.predicates.insert(is_meta_wedged, meta_wedged_continuation,
821  sst::PredicateType::ONE_TIME);
822 }

824 void ViewManager::terminate_epoch(
825  std::shared_ptr<std::map<subgroup_id_t, SubgroupSettings>>
826  next_subgroup_settings,
827  uint32_t next_num_received_size, DerechoSST& gmsSST) {
828  whenlog(logger->debug("MetaWedged is true; continuing epoch termination"););
829  // If this is the first time terminate_epoch() was called, next_view will
830  // still be null
831  bool first_call = false;
832  if(!next_view) {
833  first_call = true;
834  }
835  std::unique_lock<std::shared_timed_mutex> write_lock(view_mutex);
836  next_view = make_next_view(curr_view, gmsSST whenlog(, logger));
837  whenlog(logger->debug("Checking provisioning of view {}", next_view->vid););
838  next_subgroup_settings->clear();
839  next_num_received_size = make_subgroup_maps(subgroup_info, curr_view, *next_view, *next_subgroup_settings);
840  if(!next_view->is_adequately_provisioned) {
841  whenlog(logger->debug("Next view would not be adequately provisioned, waiting for more joins."););
842  if(first_call) {
843  // Re-register the predicates for accepting and acknowledging joins
844  register_predicates();
845  // But remove the one for start_meta_wedge
846  gmsSST.predicates.remove(leader_committed_handle);
847  }
848  // Construct a predicate that watches for any new committed change that is a
849  // join
850  int curr_num_committed = gmsSST.num_committed[curr_view->rank_of_leader()];
851  auto leader_committed_change = [this, curr_num_committed](const DerechoSST& gmsSST) {
852  return gmsSST.num_committed[curr_view->rank_of_leader()] > curr_num_committed;
853  };
854  // Construct a trigger that will re-call finish_view_change() with the same parameters
855  auto retry_next_view = [this, next_subgroup_settings,
856  next_num_received_size](DerechoSST& sst) {
857  terminate_epoch(next_subgroup_settings, next_num_received_size, sst);
858  };
859  gmsSST.predicates.insert(leader_committed_change, retry_next_view,
860  sst::PredicateType::ONE_TIME);
861  return;
862  }
863  // If execution reached here, we have a valid next view
864 
865  // go through all subgroups first and acknowledge all messages received through SST
866  for(const auto& shard_settings_pair :
867  curr_view->multicast_group->get_subgroup_settings()) {
868  const subgroup_id_t subgroup_id = shard_settings_pair.first;
869  const auto& curr_subgroup_settings = shard_settings_pair.second;
870  auto num_shard_members = curr_subgroup_settings.members.size();
871  std::vector<int> shard_senders = curr_subgroup_settings.senders;
872  auto num_shard_senders = curr_view->multicast_group->get_num_senders(shard_senders);
873  std::map<uint32_t, uint32_t> shard_ranks_by_sender_rank;
874  for(uint j = 0, l = 0; j < num_shard_members; ++j) {
875  if(shard_senders[j]) {
876  shard_ranks_by_sender_rank[l] = j;
877  l++;
878  }
879  }
880  // wait for all pending sst sends to finish
881  while(curr_view->multicast_group->check_pending_sst_sends(subgroup_id)) {
882  }
883  curr_view->gmsSST->put_with_completion();

1451  - gmsSST.num_installed[curr_view->rank_of_leader()];

1490  // unless the failed node wasn't assigned to a subgroup anyway

1492  next_unassigned_rank--;
1493  }
1494  }
1495  whenlog(logger->debug("Next view will exclude {} failed members.", leave_ranks.size()););
1496

1498  int new_rank = 0;
1499  for(int old_rank = 0; old_rank < curr_view->num_members; old_rank++) {

1502  members[new_rank] = curr_view->members[old_rank];
1503  member_ips_and_ports[new_rank] = curr_view->member_ips_and_ports[old_rank];
1504  failed[new_rank] = curr_view->failed[old_rank];
1505  ++new_rank;
1506  }
1507  }
1508

1510  int32_t my_new_rank = -1;
1511  node_id_t myID = curr_view->members[myRank];
1512  for(int i = 0; i < next_num_members; ++i) {
1513  if(members[i] == myID) {
1514  my_new_rank = i;
1515  break;
1516  }
1517  }
1518  if(my_new_rank == -1) {
1519  throw derecho_exception("Some other node reported that I failed. Node " + std::to_string(myID) + " terminating");
1520  }
1521 
1522  auto next_view = std::make_unique<View>(
1523  curr_view->vid + 1, members, member_ips_and_ports, failed, joined,
1524  departed, my_new_rank, next_unassigned_rank,
1525  curr_view->subgroup_type_order);
1526  next_view->i_know_i_am_leader = curr_view->i_know_i_am_leader;
