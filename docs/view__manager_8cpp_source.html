201 void ViewManager::await_first_view(const node_id_t my_id,
202  std::map<subgroup_id_t, SubgroupSettings>& subgroup_settings,
203  uint32_t& num_received_size) {
204  std::list<tcp::socket> waiting_join_sockets;
205  curr_view->is_adequately_provisioned = false;
206  bool joiner_failed;
207  do {
208  while(!curr_view->is_adequately_provisioned) {
209  tcp::socket client_socket = server_socket.accept();
210  node_id_t joiner_id = 0;
211  client_socket.exchange(my_id, joiner_id);
212  ip_addr& joiner_ip = client_socket.remote_ip;
213  ip_addr my_ip = client_socket.get_self_ip();
214  //Construct a new view by appending this joiner to the previous view
215  //None of these views are ever installed, so we don't use curr_view/next_view like normal
216  curr_view = std::make_unique<View>(curr_view->vid,
217  functional_append(curr_view->members, joiner_id),
218  functional_append(curr_view->member_ips, joiner_ip),
219  std::vector<char>(curr_view->num_members + 1, 0),
220  functional_append(curr_view->joined, joiner_id));
221  num_received_size = make_subgroup_maps(std::unique_ptr<View>(), *curr_view, subgroup_settings);
222  waiting_join_sockets.emplace_back(std::move(client_socket));
223  }
224  /* Now that enough joiners are queued up to make an adequate view,
225  * send it to all of them. */
226  joiner_failed = false;
227  while(!waiting_join_sockets.empty()) {
228  ip_addr& joiner_ip = waiting_join_sockets.front().remote_ip;
229  node_id_t joiner_id = curr_view->members[curr_view->rank_of(joiner_ip)];
230  auto bind_socket_write = [&waiting_join_sockets](const char* bytes, std::size_t size) {
231  bool success = waiting_join_sockets.front().write(bytes, size);
232  assert_always(success);
233  };
234  std::size_t size_of_view = mutils::bytes_size(*curr_view);
235  bool write_success = waiting_join_sockets.front().write(size_of_view);
236  if(!write_success) {
237  //The client crashed while waiting to join, so we must remove it from the view and try again
238  waiting_join_sockets.pop_front();
239  std::vector<node_id_t> filtered_members(curr_view->members.size() - 1);
240  std::vector<ip_addr> filtered_ips(curr_view->member_ips.size() - 1);
241  std::vector<node_id_t> filtered_joiners(curr_view->joined.size() - 1);
242  std::remove_copy(curr_view->members.begin(), curr_view->members.end(),
243  filtered_members.begin(), joiner_id);
244  std::remove_copy(curr_view->member_ips.begin(), curr_view->member_ips.end(),
245  filtered_ips.begin(), joiner_ip);
246  std::remove_copy(curr_view->joined.begin(), curr_view->joined.end(),
247  filtered_joiners.begin(), joiner_id);
248  curr_view = std::make_unique<View>(0, filtered_members, filtered_ips,
249  std::vector<char>(curr_view->num_members - 1, 0), filtered_joiners);
250  /* This will update curr_view->is_adequately_provisioned, so now we must
251  * start over from the beginning and test if we need to wait for more joiners. */
252  num_received_size = make_subgroup_maps(std::unique_ptr<View>(), *curr_view, subgroup_settings);
253  joiner_failed = true;
254  break;
255  }
256  mutils::post_object(bind_socket_write, *curr_view);
257  std::size_t size_of_derecho_params = mutils::bytes_size(derecho_params);
258  waiting_join_sockets.front().write(size_of_derecho_params);
259  mutils::post_object(bind_socket_write, derecho_params);
260  //Send a "0" as the size of the "old shard leaders" vector, since there are no old leaders
261  mutils::post_object(bind_socket_write, std::size_t{0});
262  waiting_join_sockets.pop_front();
263  }
264  } while(joiner_failed);
265 }

315 void ViewManager::register_predicates() {
316  /* Note that each trigger function must be wrapped in a lambda because it's
317  * a member function, and lambdas are the only way to bind "this" to a member
318  * function invocation. */
319  auto suspected_changed = [this](const DerechoSST& sst) {
320  return suspected_not_equal(sst, last_suspected);
321  };
322  auto suspected_changed_trig = [this](DerechoSST& sst) { new_suspicion(sst); };
323 
324  auto start_join_pred = [this](const DerechoSST& sst) {
325  return curr_view->i_am_leader() && has_pending_join();
326  };
327  auto start_join_trig = [this](DerechoSST& sst) { leader_start_join(sst); };
328 
329  auto change_commit_ready = [this](const DerechoSST& gmsSST) {
330  return curr_view->i_am_leader()
331  && min_acked(gmsSST, curr_view->failed) > gmsSST.num_committed[gmsSST.get_local_index()];
332  };
333  auto commit_change = [this](DerechoSST& sst) { leader_commit_change(sst); };
334 
335  auto leader_proposed_change = [this](const DerechoSST& gmsSST) {
336  return gmsSST.num_changes[curr_view->rank_of_leader()]
337  > gmsSST.num_acked[gmsSST.get_local_index()];
338  };
339  auto ack_proposed_change = [this](DerechoSST& sst) { acknowledge_proposed_change(sst); };
340 
341  auto leader_committed_changes = [this](const DerechoSST& gmsSST) {
342  return gmsSST.num_committed[curr_view->rank_of_leader()]
343  > gmsSST.num_installed[curr_view->my_rank];
344  };
345  auto view_change_trig = [this](DerechoSST& sst) { start_meta_wedge(sst); };
346 
347  if(!suspected_changed_handle.is_valid()) {
348  suspected_changed_handle = curr_view->gmsSST->predicates.insert(suspected_changed, suspected_changed_trig,
349  sst::PredicateType::RECURRENT);
350  }
351  if(!start_join_handle.is_valid()) {
352  start_join_handle = curr_view->gmsSST->predicates.insert(start_join_pred, start_join_trig,
353  sst::PredicateType::RECURRENT);
354  }
355  if(!change_commit_ready_handle.is_valid()) {
356  change_commit_ready_handle = curr_view->gmsSST->predicates.insert(change_commit_ready, commit_change,
357  sst::PredicateType::RECURRENT);
358  }
359  if(!leader_proposed_handle.is_valid()) {
360  leader_proposed_handle = curr_view->gmsSST->predicates.insert(leader_proposed_change, ack_proposed_change,
361  sst::PredicateType::RECURRENT);
362  }
363  if(!leader_committed_handle.is_valid()) {
364  leader_committed_handle = curr_view->gmsSST->predicates.insert(leader_committed_changes, view_change_trig,
365  sst::PredicateType::ONE_TIME);
366  }

371 void ViewManager::new_suspicion(DerechoSST& gmsSST) {
372  whenlog(logger->debug("Suspected[] changed");)
373  View& Vc = *curr_view;
374  int myRank = curr_view->my_rank;
375  // Aggregate suspicions into gmsSST[myRank].Suspected;
376  for(int r = 0; r < Vc.num_members; r++) {
377  for(int who = 0; who < Vc.num_members; who++) {
378  gmssst::set(gmsSST.suspected[myRank][who], gmsSST.suspected[myRank][who] || gmsSST.suspected[r][who]);
379  }
380  }
381 
382  for(int q = 0; q < Vc.num_members; q++) {
383  //If this is a new suspicion
384  if(gmsSST.suspected[myRank][q] && !Vc.failed[q]) {
385  whenlog(logger->debug("New suspicion: node {}", Vc.members[q]);)
386  //This is safer than copy_suspected, since suspected[] might change during this loop
387  last_suspected[q] = gmsSST.suspected[myRank][q];
388  if(Vc.num_failed >= (Vc.num_members + 1) / 2) {
389  throw derecho_exception("Majority of a Derecho group simultaneously failed ... shutting down");
390  }
391 
392  whenlog(logger->debug("GMS telling SST to freeze row {}", q);)
393  gmsSST.freeze(q); // Cease to accept new updates from q
394  Vc.multicast_group->wedge();
395  gmssst::set(gmsSST.wedged[myRank], true); // RDMC has halted new sends and receives in theView
396  Vc.failed[q] = true;
397  Vc.num_failed++;
398 
399  if(Vc.num_failed >= (Vc.num_members + 1) / 2) {
400  throw derecho_exception("Potential partitioning event: this node is no longer in the majority and must shut down!");
401  }
402 
403  // push change to gmsSST.suspected[myRank]
404  gmsSST.put(gmsSST.suspected.get_base() - gmsSST.getBaseAddress(), gmsSST.changes.get_base() - gmsSST.suspected.get_base());
405  // push change to gmsSST.wedged[myRank]
406  gmsSST.put(gmsSST.wedged.get_base() - gmsSST.getBaseAddress(), sizeof(gmsSST.wedged[0]));
407  if(Vc.i_am_leader() && !changes_contains(gmsSST, Vc.members[q])) // Leader initiated
408  {
409  const int next_change_index = gmsSST.num_changes[myRank] - gmsSST.num_installed[myRank];
410  if(next_change_index == (int)gmsSST.changes.size()) {
411  throw derecho_exception("Ran out of room in the pending changes list");
412  }
413 
414  gmssst::set(gmsSST.changes[myRank][next_change_index], Vc.members[q]); // Reports the failure (note that q NotIn members)
415  gmssst::increment(gmsSST.num_changes[myRank]);
416  whenlog(logger->debug("Leader proposed a change to remove failed node {}", Vc.members[q]);)
417  gmsSST.put((char*)std::addressof(gmsSST.changes[0][next_change_index]) - gmsSST.getBaseAddress(),
418  sizeof(gmsSST.changes[0][next_change_index]));
419  gmsSST.put(gmsSST.num_changes.get_base() - gmsSST.getBaseAddress(), sizeof(gmsSST.num_changes[0]));
420  }
421  }
422  }

474  * which is called "meta-wedged." To do that, this predicate trigger
475  * creates a new predicate that will fire when meta-wedged is true, and
476  * registers the next epoch termination method as its trigger.
477  */
478  auto is_meta_wedged = [this](const DerechoSST& gmsSST) {
479  for(unsigned int n = 0; n < gmsSST.get_num_rows(); ++n) {
480  if(!curr_view->failed[n] && !gmsSST.wedged[n]) {
481  return false;
482  }
483  }
484  return true;
485  };
486  auto meta_wedged_continuation = [this](DerechoSST& gmsSST) {
487  //Before the first call to terminate_epoch(), heap-allocate this map
488  auto next_subgroup_settings = std::make_shared<std::map<subgroup_id_t, SubgroupSettings>>();
489  terminate_epoch(next_subgroup_settings, 0, gmsSST);
490  };
491  gmsSST.predicates.insert(is_meta_wedged, meta_wedged_continuation, sst::PredicateType::ONE_TIME);
492 }

494 void ViewManager::terminate_epoch(std::shared_ptr<std::map<subgroup_id_t, SubgroupSettings>> next_subgroup_settings,
495  uint32_t next_num_received_size,
496  DerechoSST& gmsSST) {
497  whenlog(logger->debug("MetaWedged is true; continuing epoch termination");)
498  //If this is the first time terminate_epoch() was called, next_view will still be null
499  bool first_call = false;
500  if(!next_view) {
501  first_call = true;
502  }
503  std::unique_lock<std::shared_timed_mutex> write_lock(view_mutex);
504  next_view = make_next_view(curr_view, gmsSST whenlog(, logger));
505  whenlog(logger->debug("Checking provisioning of view {}", next_view->vid);)
506  next_subgroup_settings->clear();
507  next_num_received_size = make_subgroup_maps(curr_view, *next_view, *next_subgroup_settings);
508  if(!next_view->is_adequately_provisioned) {
509  whenlog(logger->debug("Next view would not be adequately provisioned, waiting for more joins.");)
510  if(first_call) {
511  //Re-register the predicates for accepting and acknowledging joins
512  register_predicates();
513  //But remove the one for start_meta_wedge
514  gmsSST.predicates.remove(leader_committed_handle);
515  }
516  //Construct a predicate that watches for any new committed change that is a join
517  int curr_num_committed = gmsSST.num_committed[curr_view->rank_of_leader()];
518  auto more_members_joined = [this, curr_num_committed](const DerechoSST& gmsSST) {
519  if(gmsSST.num_committed[curr_view->rank_of_leader()] > curr_num_committed) {
520  const int committed_count = gmsSST.num_committed[curr_view->rank_of_leader()]
521  - gmsSST.num_installed[curr_view->rank_of_leader()];
522  for(int change_index = curr_num_committed; change_index < committed_count; change_index++) {
523  node_id_t change_id = gmsSST.changes[curr_view->my_rank][change_index];
524  if(curr_view->rank_of(change_id) == -1) {
525  return true;
526  }
527  }
528  }
529  return false;
530  };
531  //Construct a trigger that will re-call finish_view_change() with the same parameters
532  auto retry_next_view = [this, next_subgroup_settings, next_num_received_size](DerechoSST& sst) {
533  terminate_epoch(next_subgroup_settings, next_num_received_size, sst);
534  };
535  gmsSST.predicates.insert(more_members_joined, retry_next_view, sst::PredicateType::ONE_TIME);
536  return;
537  }
538  //If execution reached here, we have a valid next view
539 
540  // go through all subgroups first and acknowledge all messages received through SST
541  for(const auto& shard_settings_pair : curr_view->multicast_group->get_subgroup_settings()) {
542  const subgroup_id_t subgroup_id = shard_settings_pair.first;
543  const auto& curr_subgroup_settings = shard_settings_pair.second;
544  auto num_shard_members = curr_subgroup_settings.members.size();
545  std::vector<int> shard_senders = curr_subgroup_settings.senders;
546  auto num_shard_senders = curr_view->multicast_group->get_num_senders(shard_senders);
547  std::map<uint32_t, uint32_t> shard_ranks_by_sender_rank;
548  for(uint j = 0, l = 0; j < num_shard_members; ++j) {
549  if(shard_senders[j]) {
550  shard_ranks_by_sender_rank[l] = j;
551  l++;
552  }
553  }
554  // wait for all pending sst sends to finish
555  while(curr_view->multicast_group->check_pending_sst_sends(subgroup_id)) {
556  }
557  curr_view->gmsSST->put_with_completion();
558  curr_view->gmsSST->sync_with_members();
559  while(curr_view->multicast_group->receiver_predicate(subgroup_id, curr_subgroup_settings, shard_ranks_by_sender_rank, num_shard_senders, *curr_view->gmsSST)) {
560  auto sst_receive_handler_lambda = [this, subgroup_id, curr_subgroup_settings,
561  shard_ranks_by_sender_rank,
562  num_shard_senders](uint32_t sender_rank,
563  volatile char* data, uint32_t size) {
564  curr_view->multicast_group->sst_receive_handler(subgroup_id, curr_subgroup_settings,
565  shard_ranks_by_sender_rank, num_shard_senders,
566  sender_rank, data, size);
567  };
568  curr_view->multicast_group->receiver_function(subgroup_id, curr_subgroup_settings,
569  shard_ranks_by_sender_rank, num_shard_senders, *curr_view->gmsSST,
570  curr_view->multicast_group->window_size, sst_receive_handler_lambda);
571  }
572  }
573 
574  curr_view->gmsSST->put_with_completion();
575  curr_view->gmsSST->sync_with_members();
576 
577  //First, for subgroups in which I'm the shard leader, do RaggedEdgeCleanup for the leader
578  auto follower_subgroups_and_shards = std::make_shared<std::map<subgroup_id_t, uint32_t>>();
579  for(const auto& shard_settings_pair : curr_view->multicast_group->get_subgroup_settings()) {
580  const subgroup_id_t subgroup_id = shard_settings_pair.first;
581  const uint32_t shard_num = shard_settings_pair.second.shard_num;
582  SubView& shard_view = curr_view->subgroup_shard_views.at(subgroup_id).at(shard_num);
583  uint num_shard_senders = 0;
584  for(auto v : shard_view.is_sender) {
585  if(v) num_shard_senders++;
586  }
587  if(num_shard_senders) {
588  if(shard_view.my_rank == curr_view->subview_rank_of_shard_leader(subgroup_id, shard_num)) {
589  leader_ragged_edge_cleanup(*curr_view, subgroup_id,
590  shard_settings_pair.second.num_received_offset,
591  shard_view.members, num_shard_senders, whenlog(logger,) next_view->members);
592  } else {
593  //Keep track of which subgroups I'm a non-leader in, and what my corresponding shard ID is
594  follower_subgroups_and_shards->emplace(subgroup_id, shard_num);
595  }
596  }
597  }
598 
599  //Wait for the shard leaders of subgroups I'm not a leader in to post global_min_ready before continuing
600  auto leader_global_mins_are_ready = [this, follower_subgroups_and_shards](const DerechoSST& gmsSST) {
601  for(const auto& subgroup_shard_pair : *follower_subgroups_and_shards) {
602  SubView& shard_view = curr_view->subgroup_shard_views.at(subgroup_shard_pair.first)
603  .at(subgroup_shard_pair.second);
604  node_id_t shard_leader = shard_view.members.at(curr_view->subview_rank_of_shard_leader(
605  subgroup_shard_pair.first, subgroup_shard_pair.second));
606  if(!gmsSST.global_min_ready[curr_view->rank_of(shard_leader)][subgroup_shard_pair.first])
607  return false;
608  }
609  return true;
610  };
611 
612  auto global_min_ready_continuation = [this, follower_subgroups_and_shards,
613  next_subgroup_settings, next_num_received_size](DerechoSST& gmsSST) {
614 
615  whenlog(logger->debug("GlobalMins are ready for all {} subgroup leaders this node is waiting on", follower_subgroups_and_shards->size());)
616  //Finish RaggedEdgeCleanup for subgroups in which I'm not the leader
617  for(const auto& subgroup_shard_pair : *follower_subgroups_and_shards) {
618  SubView& shard_view = curr_view->subgroup_shard_views.at(subgroup_shard_pair.first)
619  .at(subgroup_shard_pair.second);
620  uint num_shard_senders = 0;
621  for(auto v : shard_view.is_sender) {
622  if(v) num_shard_senders++;
623  }
624  node_id_t shard_leader = shard_view.members[curr_view->subview_rank_of_shard_leader(
625  subgroup_shard_pair.first, subgroup_shard_pair.second)];
626  follower_ragged_edge_cleanup(*curr_view, subgroup_shard_pair.first,
627  curr_view->rank_of(shard_leader),
628  curr_view->multicast_group->get_subgroup_settings()
629  .at(subgroup_shard_pair.first)
630  .num_received_offset,
631  shard_view.members,
632  num_shard_senders
633  whenlog(, logger));
634  }
635 
636  //Wait for persistence to finish for messages delivered in RaggedEdgeCleanup
637  auto persistence_finished_pred = [this](const DerechoSST& gmsSST) {
638  //For each subgroup/shard that this node is a member of...
639  for(auto subgroup_settings_pair : curr_view->multicast_group->get_subgroup_settings()) {
640  subgroup_id_t subgroup_id = subgroup_settings_pair.first;
641  if(subgroup_settings_pair.second.mode == Mode::UNORDERED) {
642  //Skip non-ordered subgroups, they never do persistence
643  continue;
644  }
645  message_id_t last_delivered_seq_num = gmsSST.delivered_num[curr_view->my_rank][subgroup_id];
646  //For each member of that shard...
647  for(const node_id_t& shard_member : subgroup_settings_pair.second.members) {
648  uint member_row = curr_view->rank_of(shard_member);
649  //Check to see if the member persisted up to the ragged edge trim
650  if(!curr_view->failed[member_row] && persistent::unpack_version<int32_t>(gmsSST.persisted_num[member_row][subgroup_id]).second < last_delivered_seq_num) {
651  return false;
652  }
653  }
654  }
655  return true;
656  };
657 
658  auto finish_view_change_trig = [this, follower_subgroups_and_shards,
659  next_subgroup_settings, next_num_received_size](DerechoSST& gmsSST) {
660  finish_view_change(follower_subgroups_and_shards, next_subgroup_settings, next_num_received_size, gmsSST);
661  };
662 
663  //Last statement in global_min_ready_continuation: register finish_view_change_trig
664  gmsSST.predicates.insert(persistence_finished_pred, finish_view_change_trig, sst::PredicateType::ONE_TIME);
665  };
666 
667  //Last statement in finish_view_change: register global_min_ready_continuation
668  gmsSST.predicates.insert(leader_global_mins_are_ready, global_min_ready_continuation, sst::PredicateType::ONE_TIME);
669 }

671 void ViewManager::finish_view_change(std::shared_ptr<std::map<subgroup_id_t, uint32_t>> follower_subgroups_and_shards,
672  std::shared_ptr<std::map<subgroup_id_t, SubgroupSettings>> next_subgroup_settings,
673  uint32_t next_num_received_size,
674  DerechoSST& gmsSST) {
675  std::unique_lock<std::shared_timed_mutex> write_lock(view_mutex);
676 
677  // Disable all the other SST predicates, except suspected_changed
678  gmsSST.predicates.remove(start_join_handle);
679  gmsSST.predicates.remove(change_commit_ready_handle);
680  gmsSST.predicates.remove(leader_proposed_handle);
681 
682  //Calculate and save the IDs of shard leaders for the old view
683  //If the old view was inadequately provisioned, this will be empty
684  //(Note: This shouldn't happen any more, as we don't allow inadequate views to be installed)
685  std::map<std::type_index, std::vector<std::vector<int64_t>>> old_shard_leaders_by_type
686  = make_shard_leaders_map(*curr_view);
687 
688  std::list<tcp::socket> joiner_sockets;
689  if(curr_view->i_am_leader() && next_view->joined.size() > 0) {
690  //If j joins have been committed, pop the next j sockets off proposed_join_sockets
691  //and send them the new View (must happen before we try to do SST setup)
692  for(std::size_t c = 0; c < next_view->joined.size(); ++c) {
693  commit_join(*next_view, proposed_join_sockets.front());
694  //save the socket for later
695  joiner_sockets.emplace_back(std::move(proposed_join_sockets.front()));
696  proposed_join_sockets.pop_front();
697  }
698  }
699 
700  // Delete the last two GMS predicates from the old SST in preparation for deleting it
701  gmsSST.predicates.remove(leader_committed_handle);
702  gmsSST.predicates.remove(suspected_changed_handle);
703 
704  node_id_t my_id = next_view->members[next_view->my_rank];
705  whenlog(logger->debug("Starting creation of new SST and DerechoGroup for view {}", next_view->vid);)
706  // if new members have joined, add their RDMA connections to SST and RDMC
707  for(std::size_t i = 0; i < next_view->joined.size(); ++i) {
708  //The new members will be the last joined.size() elements of the members lists
709  int joiner_rank = next_view->num_members - next_view->joined.size() + i;
710  rdma::impl::verbs_add_connection(next_view->members[joiner_rank], next_view->member_ips[joiner_rank],
711  my_id);
712  }
713  for(std::size_t i = 0; i < next_view->joined.size(); ++i) {
714  int joiner_rank = next_view->num_members - next_view->joined.size() + i;
715  sst::add_node(next_view->members[joiner_rank], next_view->member_ips[joiner_rank]);
716  }
717  // This will block until everyone responds to SST/RDMC initial handshakes
718  transition_multicast_group(*next_subgroup_settings, next_num_received_size);
719 
720  // Translate the old shard leaders' indices from types to new subgroup IDs
721  std::vector<std::vector<int64_t>> old_shard_leaders_by_id = translate_types_to_ids(old_shard_leaders_by_type, *next_view);
722 
723  if(curr_view->i_am_leader()) {
724  while(!joiner_sockets.empty()) {
725  //Send the array of old shard leaders, so the new member knows who to receive from
726  std::size_t size_of_vector = mutils::bytes_size(old_shard_leaders_by_id);
727  joiner_sockets.front().write(size_of_vector);
728  mutils::post_object([&joiner_sockets](const char* bytes, std::size_t size) {
729  joiner_sockets.front().write(bytes, size);
730  },
731  old_shard_leaders_by_id);
732  joiner_sockets.pop_front();
733  }
734  }
735 
736  // New members can now proceed to view_manager.start(), which will call sync()
737  next_view->gmsSST->put();
738  next_view->gmsSST->sync_with_members();
739  whenlog(logger->debug("Done setting up SST and DerechoGroup for view {}", next_view->vid);)
740  {
741  lock_guard_t old_views_lock(old_views_mutex);
742  old_views.push(std::move(curr_view));
743  old_views_cv.notify_all();
744  }
745  curr_view = std::move(next_view);
746 
747  //Write the new view to disk before using it
748  persistent::saveObject(*curr_view);
749 
750  //Re-initialize last_suspected (suspected[] has been reset to all false in the new view)
751  last_suspected.assign(curr_view->members.size(), false);
752 
753  // Register predicates in the new view
754  register_predicates();
755 
756  // First task with my new view...
757  if(curr_view->i_am_new_leader()) // I'm the new leader and everyone who hasn't failed agrees
758  {
759  curr_view->merge_changes(); // Create a combined list of Changes
760  }
761 
762  // Announce the new view to the application
763  for(auto& view_upcall : view_upcalls) {
764  view_upcall(*curr_view);
765  }
766  // One of those view upcalls is to RPCManager, which will set up TCP connections to the new members
767  // After doing that, shard leaders can send them RPC objects
768  for(subgroup_id_t subgroup_id = 0; subgroup_id < old_shard_leaders_by_id.size(); ++subgroup_id) {
769  for(uint32_t shard = 0; shard < old_shard_leaders_by_id[subgroup_id].size(); ++shard) {
770  //if I was the leader of the shard in the old view...
771  if(my_id == old_shard_leaders_by_id[subgroup_id][shard]) {
772  //send its object state to the new members
773  for(node_id_t shard_joiner : curr_view->subgroup_shard_views[subgroup_id][shard].joined) {
774  if(shard_joiner != my_id) {
775  send_subgroup_object(subgroup_id, shard_joiner);
776  }
777  }
778  }
779  }
780  }
781 
782  // Re-initialize this node's RPC objects, which includes receiving them
783  // from shard leaders if it is newly a member of a subgroup
784  whenlog(logger->debug("Initializing local Replicated Objects");)
785  initialize_subgroup_objects(my_id, *curr_view, old_shard_leaders_by_id);
786  // It's only safe to start evaluating predicates once all RPC objects exist
787  curr_view->gmsSST->start_predicate_evaluation();
788  view_change_cv.notify_all();
789 }

835 void ViewManager::receive_join(tcp::socket& client_socket) {
836  DerechoSST& gmsSST = *curr_view->gmsSST;
837  if((gmsSST.num_changes[curr_view->my_rank] - gmsSST.num_committed[curr_view->my_rank]) == (int)gmsSST.changes.size()) {

839  throw derecho_exception("Too many changes to allow a Join right now");
840  }

843  inet_aton(client_socket.remote_ip.c_str(), &joiner_ip_packed);

846  client_socket.exchange(curr_view->members[curr_view->my_rank], joining_client_id);

856  curr_view->wedge();
857  whenlog(logger->debug("Leader done wedging view.");)

872 uint32_t ViewManager::make_subgroup_maps(const std::unique_ptr<View>& prev_view,
873  View& curr_view,
874  std::map<subgroup_id_t, SubgroupSettings>& subgroup_settings) {
875  uint32_t num_received_offset = 0;
876  bool previous_was_ok = !prev_view || prev_view->is_adequately_provisioned;
877  int32_t initial_next_unassigned_rank = curr_view.next_unassigned_rank;
878  for(const auto& subgroup_type : subgroup_info.membership_function_order) {
879  subgroup_shard_layout_t subgroup_shard_views;
880  //This is the only place the subgroup membership functions are called; the results are then saved in the View
881  try {
882  auto temp = subgroup_info.subgroup_membership_functions.at(subgroup_type)(curr_view, curr_view.next_unassigned_rank, previous_was_ok);
883  //Hack to ensure RVO still works even though subgroup_shard_views had to be declared outside this scope
884  subgroup_shard_views = std::move(temp);
885  } catch(subgroup_provisioning_exception& ex) {
886  //Mark the view as inadequate and roll back everything done by previous allocation functions
887  curr_view.is_adequately_provisioned = false;
888  curr_view.next_unassigned_rank = initial_next_unassigned_rank;
889  curr_view.subgroup_shard_views.clear();
890  curr_view.subgroup_ids_by_type.clear();
891 
892  subgroup_settings.clear();
893  return 0;
894  }
895  std::size_t num_subgroups = subgroup_shard_views.size();
896  curr_view.subgroup_ids_by_type[subgroup_type] = std::vector<subgroup_id_t>(num_subgroups);

952  int myRank = curr_view->my_rank;

954  std::vector<int> join_indexes;

957  - gmsSST.num_installed[curr_view->rank_of_leader()];
958  for(int change_index = 0; change_index < committed_count; change_index++) {
959  node_id_t change_id = gmsSST.changes[myRank][change_index];
960  int change_rank = curr_view->rank_of(change_id);
961  if(change_rank != -1) {
962  //Might as well save the rank, since we'll need it again
963  leave_ranks.emplace(change_rank);
964  } else {
965  join_indexes.emplace_back(change_index);
966  }
967  }
968

970  + join_indexes.size();

972  std::vector<node_id_t> joined, members(next_num_members), departed;
973  std::vector<char> failed(next_num_members);
974  std::vector<ip_addr> member_ips(next_num_members);
975  int next_unassigned_rank = curr_view->next_unassigned_rank;
976  for(std::size_t i = 0; i < join_indexes.size(); ++i) {
977  const int join_index = join_indexes[i];
978  node_id_t joiner_id = gmsSST.changes[myRank][join_index];
979  struct in_addr joiner_ip_packed;
980  joiner_ip_packed.s_addr = gmsSST.joiner_ips[myRank][join_index];
981  char* joiner_ip_cstr = inet_ntoa(joiner_ip_packed);
982  std::string joiner_ip(joiner_ip_cstr);
983 
984  joined.emplace_back(joiner_id);
985  //New members go at the end of the members list, but it may shrink in the new view
986  int new_member_rank = curr_view->num_members - leave_ranks.size() + i;
987  members[new_member_rank] = joiner_id;
988  member_ips[new_member_rank] = joiner_ip;
989  whenlog(logger->debug("Next view will add new member with ID {}", joiner_id);)
990  }
991  for(const auto& leaver_rank : leave_ranks) {
992  departed.emplace_back(curr_view->members[leaver_rank]);
993  //Decrement next_unassigned_rank for every failure, unless the failure wasn't in a subgroup anyway
994  if(leaver_rank <= curr_view->next_unassigned_rank) {
995  next_unassigned_rank--;
996  }
997  }
998  whenlog(logger->debug("Next view will exclude {} failed members.", leave_ranks.size());)
999 
1000  //Copy member information, excluding the members that have failed
1001  int m = 0;
1002  for(int n = 0; n < curr_view->num_members; n++) {
1003  //This is why leave_ranks needs to be a set
1004  if(leave_ranks.find(n) == leave_ranks.end()) {
1005  members[m] = curr_view->members[n];
1006  member_ips[m] = curr_view->member_ips[n];
1007  failed[m] = curr_view->failed[n];
1008  ++m;
1009  }
1010  }
1011 
1012  //Initialize my_rank in next_view
1013  int32_t my_new_rank = -1;
1014  node_id_t myID = curr_view->members[myRank];
1015  for(int i = 0; i < next_num_members; ++i) {
1016  if(members[i] == myID) {
1017  my_new_rank = i;
1018  break;
1019  }
1020  }
1021  if(my_new_rank == -1) {
1022  throw derecho_exception("Some other node reported that I failed. Node " + std::to_string(myID) + " terminating");
1023  }
1024 
1025  auto next_view = std::make_unique<View>(curr_view->vid + 1, members, member_ips, failed,
1026  joined, departed, my_new_rank, next_unassigned_rank);
1027  next_view->i_know_i_am_leader = curr_view->i_know_i_am_leader;
1028  return std::move(next_view);
1029 }
