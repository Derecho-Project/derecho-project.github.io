463 
464  // if there isn't any IB device in host

466  // search for the specific device we want to work with
467  for(i = 0; i < num_devices; i++) {
468  if(!dev_name) {
469  dev_name = strdup(ibv_get_device_name(dev_list[i]));

471  if(!strcmp(ibv_get_device_name(dev_list[i]), dev_name)) {
472  ib_dev = dev_list[i];
473  break;
474  }
475  }

478  // get device handle
