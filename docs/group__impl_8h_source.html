144 template <typename FirstType, typename... RestTypes>
145 std::set<std::pair<subgroup_id_t, node_id_t>> Group<ReplicatedTypes...>::construct_objects(
146  const View& curr_view,
147  const std::unique_ptr<vector_int64_2d>& old_shard_leaders) {
148  std::set<std::pair<subgroup_id_t, uint32_t>> subgroups_to_receive;
149  if(!curr_view.is_adequately_provisioned) {
150  return subgroups_to_receive;
151  }
152  assert(replicated_objects.template get<FirstType>().empty());
153  const auto& subgroup_ids = curr_view.subgroup_ids_by_type.at(std::type_index(typeid(FirstType)));
154  for(uint32_t subgroup_index = 0; subgroup_index < subgroup_ids.size(); ++subgroup_index) {
155  subgroup_id_t subgroup_id = subgroup_ids.at(subgroup_index);
156  //Find out if this node is in any shard of this subgroup
157  bool in_subgroup = false;
158  uint32_t num_shards = curr_view.subgroup_shard_views.at(subgroup_id).size();
159  for(uint32_t shard_num = 0; shard_num < num_shards; ++shard_num) {
160  const std::vector<node_id_t>& members = curr_view.subgroup_shard_views.at(subgroup_id).at(shard_num)->members;
161  //"If this node is in subview->members for this shard"
162  if(std::find(members.begin(), members.end(), my_id) != members.end()) {
163  in_subgroup = true;
164  if(old_shard_leaders && old_shard_leaders->size() > subgroup_id
165  && (*old_shard_leaders)[subgroup_id].size() > shard_num
166  && (*old_shard_leaders)[subgroup_id][shard_num] > -1
167  && (*old_shard_leaders)[subgroup_id][shard_num] != my_id) {
168  //Construct an empty Replicated because we'll receive object state from an old leader (who is not me)
169  replicated_objects.template get<FirstType>().emplace(
170  subgroup_index, Replicated<FirstType>(my_id, subgroup_id, rpc_manager));
171  subgroups_to_receive.emplace(subgroup_id, (*old_shard_leaders)[subgroup_id][shard_num]);
172  } else {
173  replicated_objects.template get<FirstType>().emplace(
174  subgroup_index, Replicated<FirstType>(my_id, subgroup_id, rpc_manager,
175  factories.template get<FirstType>()));
176  }
177  //Store a reference to the Replicated<T> just constructed
178  objects_by_subgroup_id.emplace(subgroup_id, replicated_objects.template get<FirstType>().at(subgroup_index));
179  break; //This node can be in at most one shard, so stop here
180  }
181  }
182  if(!in_subgroup) {
183  external_callers.template get<FirstType>().emplace(subgroup_index,
184  ExternalCaller<FirstType>(my_id, subgroup_id, rpc_manager));
185  }
186  }
187  return functional_insert(subgroups_to_receive, construct_objects<RestTypes...>(curr_view, old_shard_leaders));
188 }

190 template <typename... ReplicatedTypes>
191 std::vector<RawSubgroup> Group<ReplicatedTypes...>::construct_raw_subgroups(const View& curr_view) {
192  std::vector<RawSubgroup> raw_subgroup_vector;
193  std::type_index raw_object_type(typeid(RawObject));
194  auto ids_entry = curr_view.subgroup_ids_by_type.find(raw_object_type);
195  if(ids_entry != curr_view.subgroup_ids_by_type.end()) {
196  for(uint32_t index = 0; index < ids_entry->second.size(); ++index) {
197  subgroup_id_t subgroup_id = ids_entry->second.at(index);
198  uint32_t num_shards = curr_view.subgroup_shard_views.at(subgroup_id).size();
199  bool in_subgroup = false;
200  for(uint32_t shard_num = 0; shard_num < num_shards; ++shard_num) {
201  const std::vector<node_id_t>& members = curr_view.subgroup_shard_views.at(subgroup_id).at(shard_num)->members;
202  //"If this node is in subview->members for this shard"
203  if(std::find(members.begin(), members.end(), my_id) != members.end()) {
204  in_subgroup = true;
205  raw_subgroup_vector.emplace_back(RawSubgroup(my_id, subgroup_id, view_manager));
206  break;
207  }
208  }
209  if(!in_subgroup) {
210  //Put an empty RawObject in the vector, so there's something at this index
211  raw_subgroup_vector.emplace_back(RawSubgroup(my_id, view_manager));
212  }
213  }
214  }
215  return raw_subgroup_vector;
