142 template <typename... ReplicatedTypes>
143 template <typename FirstType, typename... RestTypes>
144 std::set<std::pair<subgroup_id_t, node_id_t>> Group<ReplicatedTypes...>::construct_objects(
145  const View& curr_view,
146  const std::unique_ptr<vector_int64_2d>& old_shard_leaders) {
147  std::set<std::pair<subgroup_id_t, uint32_t>> subgroups_to_receive;
148  if(!curr_view.is_adequately_provisioned) {
149  return subgroups_to_receive;
150  }
151  const auto& subgroup_ids = curr_view.subgroup_ids_by_type.at(std::type_index(typeid(FirstType)));
152  for(uint32_t subgroup_index = 0; subgroup_index < subgroup_ids.size(); ++subgroup_index) {
153  subgroup_id_t subgroup_id = subgroup_ids.at(subgroup_index);
154  //Find out if this node is in any shard of this subgroup
155  bool in_subgroup = false;
156  uint32_t num_shards = curr_view.subgroup_shard_views.at(subgroup_id).size();
157  for(uint32_t shard_num = 0; shard_num < num_shards; ++shard_num) {
158  const std::vector<node_id_t>& members = curr_view.subgroup_shard_views.at(subgroup_id).at(shard_num).members;
159  //"If this node is in subview->members for this shard"
160  if(std::find(members.begin(), members.end(), my_id) != members.end()) {
161  in_subgroup = true;
162  //This node may have been re-assigned from a different shard, in which case we should delete the old shard's object state
163  auto old_object = replicated_objects.template get<FirstType>().find(subgroup_index);
164  if(old_object != replicated_objects.template get<FirstType>().end()
165  && old_object->second.get_shard_num() != shard_num) {
166  whenlog(logger->debug("Deleting old Replicated Object state for type {}; I was reassigned from shard {} to shard {}", typeid(FirstType).name(), old_object->second.get_shard_num(), shard_num));
167  replicated_objects.template get<FirstType>().erase(old_object);
168  }
169  //If we don't have a Replicated<T> for this (type, subgroup index), we just became a member of the shard
170  if(replicated_objects.template get<FirstType>().count(subgroup_index) == 0) {
171  //Determine if there is existing state for this shard that will need to be received
172  bool has_previous_leader = old_shard_leaders && old_shard_leaders->size() > subgroup_id
173  && (*old_shard_leaders)[subgroup_id].size() > shard_num
174  && (*old_shard_leaders)[subgroup_id][shard_num] > -1
175  && (*old_shard_leaders)[subgroup_id][shard_num] != my_id;
176  if(has_previous_leader) {
177  subgroups_to_receive.emplace(subgroup_id, (*old_shard_leaders)[subgroup_id][shard_num]);
178  }
179  if(has_previous_leader && !has_persistent_fields<FirstType>::value) {
180  /* Construct an "empty" Replicated<T>, since all of T's state will be received
181  * from the leader and there are no logs to update */
182  replicated_objects.template get<FirstType>().emplace(
183  subgroup_index, Replicated<FirstType>(my_id, subgroup_id, subgroup_index, shard_num, rpc_manager));
184  } else {
185  replicated_objects.template get<FirstType>().emplace(
186  subgroup_index, Replicated<FirstType>(my_id, subgroup_id, subgroup_index, shard_num, rpc_manager,
187  factories.template get<FirstType>()));
188  }
189  //Store a reference to the Replicated<T> just constructed
190  objects_by_subgroup_id.emplace(subgroup_id, replicated_objects.template get<FirstType>().at(subgroup_index));
191  break; //This node can be in at most one shard, so stop here
192  }
193  }
194  }
195  if(!in_subgroup) {
196  //If we have a Replicated<T> for the subgroup, but we're no longer a member, delete it
197  auto old_object = replicated_objects.template get<FirstType>().find(subgroup_index);
198  if(old_object != replicated_objects.template get<FirstType>().end()) {
199  whenlog(logger->debug("Deleting old Replicated Object state (of type {}) for subgroup {} because this node is no longer a member", typeid(FirstType).name(), subgroup_index));
200  replicated_objects.template get<FirstType>().erase(old_object);
201  objects_by_subgroup_id.erase(subgroup_id);
202  }
203  //Create an ExternalCaller for the subgroup if we don't already have one
204  external_callers.template get<FirstType>().emplace(subgroup_index,
205  ExternalCaller<FirstType>(my_id, subgroup_id, rpc_manager));
206  }
207  }
208  return functional_insert(subgroups_to_receive, construct_objects<RestTypes...>(curr_view, old_shard_leaders));
209 }

213  std::vector<RawSubgroup> raw_subgroup_vector;

216  if(ids_entry != curr_view.subgroup_ids_by_type.end()) {

223  //"If this node is in subview->members for this shard"

361 template <typename... ReplicatedTypes>
362 void Group<ReplicatedTypes...>::receive_objects(const std::set<std::pair<subgroup_id_t, node_id_t>>& subgroups_and_leaders) {
363  //This will receive one object from each shard leader in ascending order of subgroup ID
364  for(const auto& subgroup_and_leader : subgroups_and_leaders) {
365  LockedReference<std::unique_lock<std::mutex>, tcp::socket> leader_socket
366  = rpc_manager.get_socket(subgroup_and_leader.second);
367  ReplicatedObject& subgroup_object = objects_by_subgroup_id.at(subgroup_and_leader.first);
368  if(subgroup_object.is_persistent()) {
369  int64_t log_tail_length = subgroup_object.get_minimum_latest_persisted_version();
370  whenlog(logger->debug("Sending log tail length of {} for subgroup {} to node {}.", log_tail_length, subgroup_and_leader.first, subgroup_and_leader.second));
371  leader_socket.get().write(log_tail_length);
372  }
373  whenlog(logger->debug("Receiving Replicated Object state for subgroup {} from node {}", subgroup_and_leader.first, subgroup_and_leader.second));
374  std::size_t buffer_size;
375  bool success = leader_socket.get().read(buffer_size);
376  assert_always(success);
377  char buffer[buffer_size];
378  success = leader_socket.get().read(buffer, buffer_size);
379  assert_always(success);
380  subgroup_object.receive_object(buffer);
381  }
382  whenlog(logger->debug("Done receiving all Replicated Objects from subgroup leaders"));
383 }
