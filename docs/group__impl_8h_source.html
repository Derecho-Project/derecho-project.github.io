116 template <typename... ReplicatedTypes>
117 template <typename FirstType, typename... RestTypes>
118 std::set<std::pair<subgroup_id_t, node_id_t>> Group<ReplicatedTypes...>::construct_objects(
119  const View& curr_view,
120  const vector_int64_2d& old_shard_leaders) {
121  std::set<std::pair<subgroup_id_t, uint32_t>> subgroups_to_receive;
122  if(!curr_view.is_adequately_provisioned) {
123  return subgroups_to_receive;
124  }
125  //The numeric type ID of this subgroup type is its position in the ordered list of subgroup types
126  const subgroup_type_id_t subgroup_type_id = index_of_type<FirstType, ReplicatedTypes...>;
127  const auto& subgroup_ids = curr_view.subgroup_ids_by_type_id.at(subgroup_type_id);
128  for(uint32_t subgroup_index = 0; subgroup_index < subgroup_ids.size(); ++subgroup_index) {
129  subgroup_id_t subgroup_id = subgroup_ids.at(subgroup_index);
130  // Find out if this node is in any shard of this subgroup
131  bool in_subgroup = false;
132  uint32_t num_shards = curr_view.subgroup_shard_views.at(subgroup_id).size();
133  for(uint32_t shard_num = 0; shard_num < num_shards; ++shard_num) {
134  const std::vector<node_id_t>& members = curr_view.subgroup_shard_views.at(subgroup_id).at(shard_num).members;
135  //"If this node is in subview->members for this shard"
136  if(std::find(members.begin(), members.end(), my_id) != members.end()) {
137  in_subgroup = true;
138  // This node may have been re-assigned from a different shard, in which
139  // case we should delete the old shard's object state
140  auto old_object = replicated_objects.template get<FirstType>().find(subgroup_index);
141  if(old_object != replicated_objects.template get<FirstType>().end() && old_object->second.get_shard_num() != shard_num) {
142  whenlog(logger->debug("Deleting old Replicated Object state for type {}; I was reassigned from shard {} to shard {}", typeid(FirstType).name(), old_object->second.get_shard_num(), shard_num));
143  replicated_objects.template get<FirstType>().erase(old_object);
144  }
145  //If we don't have a Replicated<T> for this (type, subgroup index), we just became a member of the shard
146  if(replicated_objects.template get<FirstType>().count(subgroup_index) == 0) {
147  //Determine if there is existing state for this shard that will need to be received
148  bool has_previous_leader = old_shard_leaders.size() > subgroup_id
149  && old_shard_leaders[subgroup_id].size() > shard_num
150  && old_shard_leaders[subgroup_id][shard_num] > -1
151  && old_shard_leaders[subgroup_id][shard_num] != my_id;
152  if(has_previous_leader) {
153  subgroups_to_receive.emplace(subgroup_id, old_shard_leaders[subgroup_id][shard_num]);
154  }
155  if(has_previous_leader && !has_persistent_fields<FirstType>::value) {
156  /* Construct an "empty" Replicated<T>, since all of T's state will
157  * be received from the leader and there are no logs to update */
158  replicated_objects.template get<FirstType>().emplace(
159  subgroup_index, Replicated<FirstType>(subgroup_type_id, my_id,
160  subgroup_id, subgroup_index,
161  shard_num, rpc_manager, this));
162  } else {
163  replicated_objects.template get<FirstType>().emplace(
164  subgroup_index, Replicated<FirstType>(subgroup_type_id, my_id,
165  subgroup_id, subgroup_index, shard_num, rpc_manager,
166  factories.template get<FirstType>(), this));
167  }
168  // Store a reference to the Replicated<T> just constructed
169  objects_by_subgroup_id.emplace(subgroup_id,
170  replicated_objects.template get<FirstType>().at(subgroup_index));
171  break; // This node can be in at most one shard, so stop here
172  }
173  }
174  }
175  if(!in_subgroup) {
176  // If we have a Replicated<T> for the subgroup, but we're no longer a
177  // member, delete it
178  auto old_object = replicated_objects.template get<FirstType>().find(subgroup_index);
179  if(old_object != replicated_objects.template get<FirstType>().end()) {
180  whenlog(logger->debug("Deleting old Replicated Object state (of type {}) for subgroup {} because this node is no longer a member", typeid(FirstType).name(), subgroup_index));
181  replicated_objects.template get<FirstType>().erase(old_object);
182  objects_by_subgroup_id.erase(subgroup_id);
183  }
184  // Create an ExternalCaller for the subgroup if we don't already have one
185  external_callers.template get<FirstType>().emplace(
186  subgroup_index, ExternalCaller<FirstType>(subgroup_type_id,
187  my_id, subgroup_id, rpc_manager));
188  }
189  }
190  return functional_insert(subgroups_to_receive, construct_objects<RestTypes...>(curr_view, old_shard_leaders));
191 }

244 template <typename... ReplicatedTypes>
245 void Group<ReplicatedTypes...>::update_tcp_connections_callback(const View& new_view) {
246  if(std::find(new_view.joined.begin(), new_view.joined.end(), my_id) != new_view.joined.end()) {
247  //If this node is in the joined list, we need to set up a connection to everyone
248  for(int i = 0; i < new_view.num_members; ++i) {
249  if(new_view.members[i] != my_id) {
250  tcp_sockets->add_node(new_view.members[i], {std::get<0>(new_view.member_ips_and_ports[i]), std::get<PORT_TYPE::RPC>(new_view.member_ips_and_ports[i])});
251  whendebug(logger->debug("Established a TCP connection to node {}", new_view.members[i]);)
252  }
253  }
254  } else {
255  //This node is already a member, so we already have connections to the previous view's members
256  for(const node_id_t& joiner_id : new_view.joined) {
257  tcp_sockets->add_node(joiner_id,
258  {std::get<0>(new_view.member_ips_and_ports[new_view.rank_of(joiner_id)]),
259  std::get<PORT_TYPE::RPC>(new_view.member_ips_and_ports[new_view.rank_of(joiner_id)])});
260  whenlog(logger->debug("Established a TCP connection to node {}", joiner_id););
261  }
262  for(const node_id_t& removed_id : new_view.departed) {
263  whenlog(logger->debug("Removing TCP connection for failed node {}", removed_id););
264  tcp_sockets->delete_node(removed_id);
265  }
266  }

325 template <typename... ReplicatedTypes>
326 void Group<ReplicatedTypes...>::receive_objects(const std::set<std::pair<subgroup_id_t, node_id_t>>& subgroups_and_leaders) {
327  //This will receive one object from each shard leader in ascending order of subgroup ID
328  for(const auto& subgroup_and_leader : subgroups_and_leaders) {
329  LockedReference<std::unique_lock<std::mutex>, tcp::socket> leader_socket
330  = tcp_sockets->get_socket(subgroup_and_leader.second);
331  ReplicatedObject& subgroup_object = objects_by_subgroup_id.at(subgroup_and_leader.first);
332  if(subgroup_object.is_persistent()) {
333  int64_t log_tail_length = subgroup_object.get_minimum_latest_persisted_version();
334  whenlog(logger->debug("Sending log tail length of {} for subgroup {} to node {}.", log_tail_length, subgroup_and_leader.first, subgroup_and_leader.second));
335  leader_socket.get().write(log_tail_length);
336  }
337  whenlog(logger->debug("Receiving Replicated Object state for subgroup {} from node {}", subgroup_and_leader.first, subgroup_and_leader.second));
338  std::size_t buffer_size;
339  bool success = leader_socket.get().read(buffer_size);
340  assert_always(success);
341  char buffer[buffer_size];
342  success = leader_socket.get().read(buffer, buffer_size);
343  assert_always(success);
344  subgroup_object.receive_object(buffer);
345  }
346  whenlog(logger->debug("Done receiving all Replicated Objects from subgroup leaders"));
347 }
