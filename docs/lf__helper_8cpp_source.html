583 static void polling_loop() {
584  pthread_setname_np(pthread_self(), "rdmc_poll");
585 
586  const int max_cq_entries = 1024;
587  unique_ptr<fi_cq_data_entry[]> cq_entries(new fi_cq_data_entry[max_cq_entries]);
588 
589  while (true) {
590  int num_completions = 0;
591  while (num_completions == 0 || num_completions == -FI_EAGAIN) {
592  if (polling_loop_shutdown_flag) return;
593  uint64_t poll_end = get_time() + (interrupt_mode ? 0L : 50000000L);
594  do {
595  if(polling_loop_shutdown_flag) return;
596  num_completions = fi_cq_read(g_ctxt.cq, cq_entries.get(), max_cq_entries);
597  } while(num_completions == 0 && get_time() < poll_end);
598 
599  if (num_completions == 0) {
602  num_completions = fi_cq_read(g_ctxt.cq, cq_entries.get(), max_cq_entries);
603 
604  if (num_completions == 0) {
605  pollfd file_descriptor;
606  fi_control(&g_ctxt.cq->fid, FI_GETWAIT, &file_descriptor);
607  int rc = 0;
608  while (rc == 0 && !polling_loop_shutdown_flag) {
609  if(polling_loop_shutdown_flag) return;
610  rc = poll(&file_descriptor, 1, 50);
611  }
612 
613  if (rc > 0) {
614  num_completions = fi_cq_read(g_ctxt.cq, cq_entries.get(), max_cq_entries);
615  }
616  }

631  continue;

633  uint64_t masked_wr_id = (uint64_t)cq_entry.op_context & 0x0000ffffffffffffull;
634  uint32_t opcode = (uint32_t)EXTRACT_RDMA_OP_CODE(cq_entry.op_context);
635  uint32_t immediate = cq_entry.data;
636  if (type >= completion_handlers.size()) {
637  // Unrecognized message type
638  } else if (opcode == RDMA_OP_SEND) {
639  completion_handlers[type].send(masked_wr_id, immediate,
640  cq_entry.len);
641  } else if (opcode == RDMA_OP_RECV) {
642  completion_handlers[type].recv(masked_wr_id, immediate,
643  cq_entry.len);
644  } else if (opcode == RDMA_OP_WRITE) {
645  completion_handlers[type].write(masked_wr_id, immediate,
646  cq_entry.len);
647  } else {
648  puts("Sent unrecognized completion type?!");
649  }
650  }
651  }
