97  const std::decay_t<Args>&... remote_args) {
98  auto invocation_id = mutils::long_rand();
99  std::size_t size = mutils::bytes_size(invocation_id);
100  {
101  auto t = {std::size_t{0}, std::size_t{0}, mutils::bytes_size(remote_args)...};
102  size += std::accumulate(t.begin(), t.end(), 0);
103  }
104  char* serialized_args = out_alloc(size);
105  {
106  auto v = serialized_args + mutils::to_bytes(invocation_id, serialized_args);
107  auto check_size = mutils::bytes_size(invocation_id) + serialize_all(v, remote_args...);
108  assert_always(check_size == size);
109  }

112  // default-initialize the maps
113  PendingResults<Ret>& pending_results = results_map[invocation_id];

245  long int invocation_id = ((long int*)_recv_buf)[0];
