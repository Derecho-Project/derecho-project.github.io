61 void DefaultSubgroupAllocator::assign_subgroup(const View& curr_view, int& next_unassigned_rank, const ShardAllocationPolicy& subgroup_policy) {
62  if(subgroup_policy.even_shards) {
63  if(static_cast<int>(curr_view.members.size()) - next_unassigned_rank
64  < subgroup_policy.num_shards * subgroup_policy.nodes_per_shard) {
65  throw subgroup_provisioning_exception();
66  }
67  }
68  previous_assignment->emplace_back(std::vector<SubView>());
69  for(int shard_num = 0; shard_num < subgroup_policy.num_shards; ++shard_num) {
70  if(!subgroup_policy.even_shards && next_unassigned_rank + subgroup_policy.num_nodes_by_shard[shard_num] >= (int)curr_view.members.size()) {
71  throw subgroup_provisioning_exception();
72  }
73  int nodes_needed = subgroup_policy.even_shards ? subgroup_policy.nodes_per_shard : subgroup_policy.num_nodes_by_shard[shard_num];
74  std::vector<node_id_t> desired_nodes(&curr_view.members[next_unassigned_rank],
75  &curr_view.members[next_unassigned_rank + nodes_needed]);
76  next_unassigned_rank += nodes_needed;
77  Mode delivery_mode = subgroup_policy.even_shards ? subgroup_policy.shards_mode : subgroup_policy.modes_by_shard[shard_num];
78  (*previous_assignment).back().emplace_back(curr_view.make_subview(desired_nodes, delivery_mode));
79  }

82 subgroup_shard_layout_t DefaultSubgroupAllocator::operator()(const View& curr_view,
83  int& next_unassigned_rank,
84  bool previous_was_successful) {
85  if(previous_was_successful) {
86  //Save the previous assignment since it was successful
87  last_good_assignment = deep_pointer_copy(previous_assignment);
88  } else {
89  //Overwrite previous_assignment with the one before that, if it exists
90  previous_assignment = deep_pointer_copy(last_good_assignment);
91  }
92  if(previous_assignment) {
93  for(int subgroup_num = 0; subgroup_num < policy.num_subgroups; ++subgroup_num) {
94  int num_shards_in_subgroup;
95  if(policy.identical_subgroups) {
96  num_shards_in_subgroup = policy.shard_policy_by_subgroup[0].num_shards;
97  } else {
98  num_shards_in_subgroup = policy.shard_policy_by_subgroup[subgroup_num].num_shards;
99  }
100  for(int shard_num = 0; shard_num < num_shards_in_subgroup; ++shard_num) {
101  //Check each member of the shard in the previous assignment
102  for(std::size_t shard_rank = 0;
103  shard_rank < (*previous_assignment)[subgroup_num][shard_num].members.size();
104  ++shard_rank) {
105  if(curr_view.rank_of((*previous_assignment)[subgroup_num][shard_num].members[shard_rank]) == -1) {
106  //This node is not in the current view, so take the next available one
107  if(next_unassigned_rank >= (int)curr_view.members.size()) {
108  throw subgroup_provisioning_exception();
109  }
110  (*previous_assignment)[subgroup_num][shard_num].members[shard_rank] = curr_view.members[next_unassigned_rank];
111  (*previous_assignment)[subgroup_num][shard_num].member_ips[shard_rank] = curr_view.member_ips[next_unassigned_rank];
112  next_unassigned_rank++;
113  } else {
114  }
115  }
116  //These will be initialized from scratch by the calling ViewManager
117  (*previous_assignment)[subgroup_num][shard_num].joined.clear();
118  (*previous_assignment)[subgroup_num][shard_num].departed.clear();
119  }
120  }
121  } else {
122  previous_assignment = std::make_unique<subgroup_shard_layout_t>();
123  for(int subgroup_num = 0; subgroup_num < policy.num_subgroups; ++subgroup_num) {
124  if(policy.identical_subgroups) {
125  assign_subgroup(curr_view, next_unassigned_rank, policy.shard_policy_by_subgroup[0]);
126  } else {
127  assign_subgroup(curr_view, next_unassigned_rank, policy.shard_policy_by_subgroup[subgroup_num]);
128  }
129  }
130  }
131  return *previous_assignment;
132 }

134 subgroup_shard_layout_t CrossProductAllocator::operator()(const View& curr_view,
135  int& next_unassigned_rank,
136  bool previous_was_successful) {
137  /* Ignore previous_was_successful and next_unassigned_rank, because this subgroup's assignment
138  * is based entirely on the source and target subgroups, and doesn't provision any new nodes. */
139  subgroup_id_t source_subgroup_id = curr_view.subgroup_ids_by_type.at(policy.source_subgroup.first)
140  .at(policy.source_subgroup.second);
141  subgroup_id_t target_subgroup_id = curr_view.subgroup_ids_by_type.at(policy.target_subgroup.first)
142  .at(policy.target_subgroup.second);
143  int num_source_members = 0;
144  for(const auto& shard_view : curr_view.subgroup_shard_views[source_subgroup_id]) {
145  num_source_members += shard_view.members.size();
146  }
147  int num_target_shards = curr_view.subgroup_shard_views[target_subgroup_id].size();
148  //Each subgroup will have only one shard, since they'll all overlap, so there are source * target subgroups
149  subgroup_shard_layout_t assignment(num_source_members * num_target_shards);
150  //I want a list of all members of the source subgroup, "flattened" out of shards, but we don't have that
151  //Instead, iterate through the source's shards in order and keep a consistent index
152  int source_member_index = 0;
153  for(std::size_t source_shard = 0; source_shard < curr_view.subgroup_shard_views[source_subgroup_id].size(); ++source_shard) {
154  for(const auto& source_node : curr_view.subgroup_shard_views[source_subgroup_id][source_shard].members) {
155  for(int target_shard = 0; target_shard < num_target_shards; ++target_shard) {
156  const SubView& target_shard_view = curr_view.subgroup_shard_views[target_subgroup_id][target_shard];
157  std::vector<node_id_t> desired_nodes(target_shard_view.members.size() + 1);
158  desired_nodes[0] = source_node;
159  std::copy(target_shard_view.members.begin(),
160  target_shard_view.members.end(),
161  desired_nodes.begin() + 1);
162  std::vector<int> sender_flags(desired_nodes.size(), false);
163  sender_flags[0] = true;
164  //The vector at this subgroup's index will be default initialized, so push_back a single shard
165  assignment[source_member_index * num_target_shards + target_shard].push_back(
166  curr_view.make_subview(desired_nodes, Mode::ORDERED, sender_flags));
167  //Now, to send from source_member_index to target_shard, we can use the subgroup at
168  //source_member_index * num_target_shards + target_shard
169  }
170  source_member_index++;
171  }
172  }
173  return assignment;
174 }
