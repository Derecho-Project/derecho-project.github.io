65 std::vector<SubView> DefaultSubgroupAllocator::assign_subgroup(const std::unique_ptr<View>& prev_view,
66  View& curr_view,
67  const ShardAllocationPolicy& subgroup_policy) const {
68  if(subgroup_policy.even_shards) {
69  if(static_cast<int>(curr_view.members.size()) - curr_view.next_unassigned_rank
70  < subgroup_policy.num_shards * subgroup_policy.nodes_per_shard) {
71  throw subgroup_provisioning_exception();
72  }
73  }
74  std::vector<SubView> subgroup_allocation;
75  for(int shard_num = 0; shard_num < subgroup_policy.num_shards; ++shard_num) {
76  if(!subgroup_policy.even_shards
77  && curr_view.next_unassigned_rank + subgroup_policy.num_nodes_by_shard[shard_num]
78  >= (int)curr_view.members.size()) {
79  throw subgroup_provisioning_exception();
80  }
81  int nodes_needed = subgroup_policy.even_shards ? subgroup_policy.nodes_per_shard : subgroup_policy.num_nodes_by_shard[shard_num];
82  std::vector<node_id_t> desired_nodes(&curr_view.members[curr_view.next_unassigned_rank],
83  &curr_view.members[curr_view.next_unassigned_rank + nodes_needed]);
84  curr_view.next_unassigned_rank += nodes_needed;
85  Mode delivery_mode = subgroup_policy.even_shards ? subgroup_policy.shards_mode : subgroup_policy.modes_by_shard[shard_num];
86  subgroup_allocation.emplace_back(curr_view.make_subview(desired_nodes, delivery_mode));
87  }
88  return subgroup_allocation;
89 }

91 subgroup_shard_layout_t DefaultSubgroupAllocator::compute_standard_membership(const std::type_index& subgroup_type,
92  const std::unique_ptr<View>& prev_view,
93  View& curr_view) const {
94  const SubgroupAllocationPolicy& subgroup_policy = std::get<SubgroupAllocationPolicy>(policies.at(subgroup_type));
95  if(prev_view) {
96  const uint32_t subgroup_type_id = index_of(prev_view->subgroup_type_order, subgroup_type);
97  /* Subgroups of the same type will have contiguous IDs because they were created in order.
98  * So the previous assignment is the slice of the previous subgroup_shard_views vector
99  * starting at the first subgroup's ID, and extending for num_subgroups entries.
100  */
101  const std::size_t previous_assignment_offset = prev_view->subgroup_ids_by_type_id.at(subgroup_type_id)[0];
102  subgroup_shard_layout_t next_assignment(&prev_view->subgroup_shard_views[previous_assignment_offset],
103  &prev_view->subgroup_shard_views[previous_assignment_offset
104  + subgroup_policy.num_subgroups]);
105  for(int subgroup_num = 0; subgroup_num < subgroup_policy.num_subgroups; ++subgroup_num) {
106  int num_shards_in_subgroup;
107  if(subgroup_policy.identical_subgroups) {
108  num_shards_in_subgroup = subgroup_policy.shard_policy_by_subgroup[0].num_shards;
109  } else {
110  num_shards_in_subgroup = subgroup_policy.shard_policy_by_subgroup[subgroup_num].num_shards;
111  }
112  for(int shard_num = 0; shard_num < num_shards_in_subgroup; ++shard_num) {
113  //Check each member of the shard in the previous assignment
114  const std::vector<node_id_t>& previous_shard_members = prev_view->subgroup_shard_views[previous_assignment_offset + subgroup_num][shard_num].members;
115  for(std::size_t shard_rank = 0; shard_rank < previous_shard_members.size(); ++shard_rank) {
116  if(curr_view.rank_of(previous_shard_members[shard_rank]) == -1) {
117  //That node is not in the current view, so take the next available one
118  if(curr_view.next_unassigned_rank >= static_cast<int>(curr_view.members.size())) {
119  throw subgroup_provisioning_exception();
120  }
121  next_assignment[subgroup_num][shard_num].members[shard_rank] = curr_view.members[curr_view.next_unassigned_rank];
122  next_assignment[subgroup_num][shard_num].member_ips_and_ports[shard_rank] = curr_view.member_ips_and_ports[curr_view.next_unassigned_rank];
123  curr_view.next_unassigned_rank++;
124  }
125  }
126  //These will be initialized from scratch by the calling ViewManager, so don't keep the copied values
127  next_assignment[subgroup_num][shard_num].joined.clear();
128  next_assignment[subgroup_num][shard_num].departed.clear();
129  }
130  }
131  return next_assignment;
132  } else {
133  subgroup_shard_layout_t next_assignment;
134  for(int subgroup_num = 0; subgroup_num < subgroup_policy.num_subgroups; ++subgroup_num) {
135  if(subgroup_policy.identical_subgroups) {
136  next_assignment.push_back(assign_subgroup(prev_view, curr_view,
137  subgroup_policy.shard_policy_by_subgroup[0]));
138  } else {
139  next_assignment.push_back(assign_subgroup(prev_view, curr_view,
140  subgroup_policy.shard_policy_by_subgroup[subgroup_num]));
141  }
142  }
143  return next_assignment;
144  }
145 }

147 subgroup_shard_layout_t DefaultSubgroupAllocator::compute_cross_product_membership(const std::type_index& subgroup_type,
148  const std::unique_ptr<View>& prev_view,
149  View& curr_view) const {
150  const CrossProductPolicy& cross_product_policy = std::get<CrossProductPolicy>(policies.at(subgroup_type));
151  /* Ignore prev_view and next_unassigned_rank, because this subgroup's assignment is based
152  * entirely on the source and target subgroups, and doesn't provision any new nodes. */
153  subgroup_type_id_t source_subgroup_type = index_of(curr_view.subgroup_type_order,
154  cross_product_policy.source_subgroup.first);
155  subgroup_type_id_t target_subgroup_type = index_of(curr_view.subgroup_type_order,
156  cross_product_policy.target_subgroup.first);
157  subgroup_id_t source_subgroup_id = curr_view.subgroup_ids_by_type_id.at(source_subgroup_type)
158  .at(cross_product_policy.source_subgroup.second);
159  subgroup_id_t target_subgroup_id = curr_view.subgroup_ids_by_type_id.at(target_subgroup_type)
160  .at(cross_product_policy.target_subgroup.second);
161  int num_source_members = 0;
162  for(const auto& shard_view : curr_view.subgroup_shard_views[source_subgroup_id]) {
163  num_source_members += shard_view.members.size();
164  }
165  int num_target_shards = curr_view.subgroup_shard_views[target_subgroup_id].size();
166  //Each subgroup will have only one shard, since they'll all overlap, so there are source * target subgroups
167  subgroup_shard_layout_t assignment(num_source_members * num_target_shards);
168  //I want a list of all members of the source subgroup, "flattened" out of shards, but we don't have that
169  //Instead, iterate through the source's shards in order and keep a consistent index
170  int source_member_index = 0;
171  for(std::size_t source_shard = 0;
172  source_shard < curr_view.subgroup_shard_views[source_subgroup_id].size();
173  ++source_shard) {
174  for(const auto& source_node : curr_view.subgroup_shard_views[source_subgroup_id][source_shard].members) {
175  for(int target_shard = 0; target_shard < num_target_shards; ++target_shard) {
176  const SubView& target_shard_view = curr_view.subgroup_shard_views[target_subgroup_id][target_shard];
177  std::vector<node_id_t> desired_nodes(target_shard_view.members.size() + 1);
178  desired_nodes[0] = source_node;
179  std::copy(target_shard_view.members.begin(),
180  target_shard_view.members.end(),
181  desired_nodes.begin() + 1);
182  std::vector<int> sender_flags(desired_nodes.size(), false);
183  sender_flags[0] = true;
184  //The vector at this subgroup's index will be default initialized, so push_back a single shard
185  assignment[source_member_index * num_target_shards + target_shard].push_back(
186  curr_view.make_subview(desired_nodes, Mode::ORDERED, sender_flags));
187  //Now, to send from source_member_index to target_shard, we can use the subgroup at
188  //source_member_index * num_target_shards + target_shard
189  }
190  source_member_index++;
191  }
192  }
193  return assignment;
194 }
