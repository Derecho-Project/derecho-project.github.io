811 void test_pattern() {
812  size_t n = 0;
813  auto t = get_time();
814  for(size_t group_size = 2; group_size <= 64; group_size++) {
815  for(size_t message_size = 1; message_size <= 32; message_size++) {
816  size_t total_steps = message_size + ceil(log2(group_size)) - 1;
817  for(unsigned int step = 0; step < total_steps; step++) {
818  for(unsigned int node = 0; node < group_size; node++) {
819  // Compute the outgoing transfer for this node/step
820  auto transfer = binomial_schedule::get_outgoing_transfer(
821  node, step, group_size, floor(log2(group_size)),
822  message_size, total_steps);
823  n++;
824 
825  if(transfer) {
826  // See what the supposed sender is doing this step
827  auto reverse = binomial_schedule::get_incoming_transfer(
828  transfer->target, step, group_size,
829  floor(log2(group_size)), message_size, total_steps);
830  n++;
831 
832  // Make sure the two nodes agree
833  if(!reverse) throw false;
834  if(transfer->block_number != reverse->block_number)
835  throw false;
836 
837  // If we aren't the root sender, also check that the
838  // node got this block on a past step.
839  if(node != 0) {
840  for(int s = step - 1; s >= 0; s--) {
841  auto prev =
842  binomial_schedule::get_incoming_transfer(
843  node, s, group_size,
844  floor(log2(group_size)), message_size,
845  total_steps);
846  n++;
847  if(prev &&
848  prev->block_number == transfer->block_number)
849  break;
850 
851  if(s == 0) {
852  throw false;
853  }
854  }
855  }
856  }
857 
858  // Compute the incoming transfer for this node/step
859  transfer = binomial_schedule::get_incoming_transfer(
860  node, step, group_size, floor(log2(group_size)),
861  message_size, total_steps);
862  n++;
863 
864  if(transfer) {
865  // Again make sure the supposed receiver agrees
866  auto reverse = binomial_schedule::get_outgoing_transfer(
867  transfer->target, step, group_size,
868  floor(log2(group_size)), message_size, total_steps);
869  n++;
870  if(!reverse) throw false;
871  if(transfer->block_number != reverse->block_number)
872  throw false;
873  if(reverse->target != node) throw false;
874 
875  // Make sure we don't already have the block we're
876  // getting.
877  for(int s = step - 1; s >= 0; s--) {
878  auto prev = binomial_schedule::get_incoming_transfer(
879  node, s, group_size, floor(log2(group_size)),
880  message_size, total_steps);
881  n++;
882  if(prev &&
883  prev->block_number == transfer->block_number) {
884  throw false;
885  }
886  }
887  }
888  }
889  }
890 
891  // Make sure that all nodes get every block
892  for(unsigned int node = 1; node < group_size; node++) {
893  set<size_t> blocks;
894  for(unsigned int step = 0; step < total_steps; step++) {
895  auto transfer = binomial_schedule::get_incoming_transfer(
896  node, step, group_size, floor(log2(group_size)),
897  message_size, total_steps);
898  n++;
899 
900  if(transfer) blocks.insert(transfer->block_number);
901  }
902  if(blocks.size() != message_size) throw false;
903  }
904  }
905  }
906  auto diff = get_time() - t;
907  printf("average time = %f ns\n", (double)diff / n);
908  puts("PASS");
909 }
