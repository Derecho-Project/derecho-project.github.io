454  if(msg.size > 0) {
455  char* buf = msg.message_buffer.buffer.get();
456  header* h = (header*)(buf);
457  // cooked send
458  if(h->cooked_send) {
459  buf += h->header_size;
460  auto payload_size = msg.size - h->header_size;
461  rpc_callback(subgroup_num, msg.sender_id, buf, payload_size);
462  }
463  // raw send
464  else {
465  callbacks.global_stability_callback(subgroup_num, msg.sender_id, msg.index,
466  buf + h->header_size, msg.size);
467  }
468  if(file_writer) {
469  persistence::message msg_for_filewriter{buf + h->header_size,
470  msg.size, (uint32_t)sst->vid[member_index],
471  msg.sender_id, (uint64_t)msg.index,
472  h->cooked_send};
473  //the sequence number needs to use the sender's within-shard rank, not its ID
474  auto& shard_members = subgroup_to_membership.at(subgroup_num);
475  std::vector<int> shard_senders = subgroup_to_senders_and_sender_rank.at(subgroup_num).first;
476  auto num_shard_senders = get_num_senders(shard_senders);
477  uint shard_rank;
478  uint sender_rank = 0;
479  for(shard_rank = 0; shard_rank < shard_members.size(); ++shard_rank) {
480  if(shard_members[shard_rank] == msg.sender_id) break;
481  if(shard_senders[shard_rank])

483  }

485  non_persistent_messages[subgroup_num].emplace(sequence_number, std::move(msg));
486  file_writer->write_message(msg_for_filewriter);

516  uint32_t shard_num, shard_index;
517  std::tie(shard_num, shard_index) = p.second;
518  std::vector<node_id_t> shard_members = subgroup_to_membership.at(subgroup_num);
519  auto num_shard_members = shard_members.size();
520  auto stability_pred = [this](
521  const DerechoSST& sst) { return true; };
522  auto stability_trig =
523  [this, subgroup_num, shard_members, num_shard_members](DerechoSST& sst) {
524  // compute the min of the seq_num
525  long long int min_seq_num = sst.seq_num[node_id_to_sst_index[shard_members[0]]][subgroup_num];
526  for(uint i = 0; i < num_shard_members; ++i) {
527  if(sst.seq_num[node_id_to_sst_index[shard_members[i]]][subgroup_num] < min_seq_num) {
528  min_seq_num = sst.seq_num[node_id_to_sst_index[shard_members[i]]][subgroup_num];
529  }
530  }
531  if(min_seq_num > sst.stable_num[member_index][subgroup_num]) {
532  logger->debug("Subgroup {}, updating stable_num to {}", subgroup_num, min_seq_num);
533  sst.stable_num[member_index][subgroup_num] = min_seq_num;
534  sst.put(get_shard_sst_indices(subgroup_num),
535  (char*)std::addressof(sst.stable_num[0][subgroup_num]) - sst.getBaseAddress(),
536  sizeof(long long int));
537  }
538  };
539  stability_pred_handle = sst->predicates.insert(
540  stability_pred, stability_trig, sst::PredicateType::RECURRENT);
541 
542  auto delivery_pred = [this](
543  const DerechoSST& sst) { return true; };
544  auto delivery_trig = [this, subgroup_num, shard_members, num_shard_members](
545  DerechoSST& sst) {
546  std::lock_guard<std::mutex> lock(msg_state_mtx);
547  // compute the min of the stable_num
548  long long int min_stable_num = sst.stable_num[node_id_to_sst_index[shard_members[0]]][subgroup_num];
549  for(uint i = 0; i < num_shard_members; ++i) {
550  if(sst.stable_num[node_id_to_sst_index[shard_members[i]]][subgroup_num] < min_stable_num) {
551  min_stable_num = sst.stable_num[node_id_to_sst_index[shard_members[i]]][subgroup_num];
552  }
553  }
554 
555  if(!locally_stable_messages[subgroup_num].empty()) {
556  long long int least_undelivered_seq_num = locally_stable_messages[subgroup_num].begin()->first;
557  if(least_undelivered_seq_num <= min_stable_num) {
558  logger->debug("Subgroup {}, can deliver a locally stable message: min_stable_num={} and least_undelivered_seq_num={}", subgroup_num, min_stable_num, least_undelivered_seq_num);
559  Message& msg = locally_stable_messages[subgroup_num].begin()->second;
560  deliver_message(msg, subgroup_num);
561  sst.delivered_num[member_index][subgroup_num] = least_undelivered_seq_num;
562  sst.put(get_shard_sst_indices(subgroup_num),
563  (char*)std::addressof(sst.delivered_num[0][subgroup_num]) - sst.getBaseAddress(),
564  sizeof(long long int));
565  locally_stable_messages[subgroup_num].erase(locally_stable_messages[subgroup_num].begin());
566  }
567  }
568  };
569 
570  delivery_pred_handle = sst->predicates.insert(delivery_pred, delivery_trig, sst::PredicateType::RECURRENT);
571 
572  uint32_t num_shard_senders;
573  int shard_sender_index;
574  std::vector<int> shard_senders;
575  std::tie(shard_senders, shard_sender_index) = subgroup_to_senders_and_sender_rank.at(subgroup_num);
576  num_shard_senders = get_num_senders(shard_senders);
577  if(shard_sender_index >= 0) {
578  auto sender_pred = [this, subgroup_num, shard_members, num_shard_members, shard_sender_index, num_shard_senders](const DerechoSST& sst) {
579  long long int seq_num = next_message_to_deliver[subgroup_num] * num_shard_senders + shard_sender_index;
580  for(uint i = 0; i < num_shard_members; ++i) {
581  if(sst.delivered_num[node_id_to_sst_index[shard_members[i]]][subgroup_num] < seq_num
582  || (file_writer && sst.persisted_num[node_id_to_sst_index[shard_members[i]]][subgroup_num] < seq_num)) {
583  return false;
584  }
585  }
586  return true;
587  };
588  auto sender_trig = [this, subgroup_num](DerechoSST& sst) {
589  sender_cv.notify_all();
590  next_message_to_deliver[subgroup_num]++;
591  };
592  sender_pred_handle = sst->predicates.insert(sender_pred, sender_trig,
593  sst::PredicateType::RECURRENT);
594  }
595  }
596 }

635 void MulticastGroup::send_loop() {
636  subgroup_id_t subgroup_to_send = 0;
637  auto should_send_to_subgroup = [&](subgroup_id_t subgroup_num) {
638  if(!rdmc_groups_created) {
639  return false;
640  }
641  if(pending_sends[subgroup_num].empty()) {
642  return false;
643  }
644  Message& msg = pending_sends[subgroup_num].front();
645  uint32_t shard_num;
646  uint32_t shard_index;
647 
648  std::tie(shard_num, shard_index) = subgroup_to_shard_and_rank.at(subgroup_num);
649  uint32_t num_shard_senders;
650  int shard_sender_index;
651  std::vector<int> shard_senders;
652  std::tie(shard_senders, shard_sender_index) = subgroup_to_senders_and_sender_rank.at(subgroup_num);
653  num_shard_senders = get_num_senders(shard_senders);
654  assert(shard_sender_index >= 0);
655 
656  // std::cout << "num_received offset = " << subgroup_to_num_received_offset.at(subgroup_num) + shard_sender_index <<
657  // ", num_received entry " << sst->num_received[member_index][subgroup_to_num_received_offset.at(subgroup_num) + shard_sender_index] <<
658  // ", message index = " << msg.index << std::endl;
659  if(sst->num_received[member_index][subgroup_to_num_received_offset.at(subgroup_num) + shard_sender_index] < msg.index - 1) {
660  return false;
661  }
662 
663  std::vector<node_id_t> shard_members = subgroup_to_membership.at(subgroup_num);
664  auto num_shard_members = shard_members.size();
665  assert(num_shard_members >= 1);
666  for(uint i = 0; i < num_shard_members; ++i) {
667  if(sst->delivered_num[node_id_to_sst_index[shard_members[i]]][subgroup_num] < (int)((msg.index - window_size) * num_shard_senders + shard_sender_index)
668  || (file_writer && sst->persisted_num[node_id_to_sst_index[shard_members[i]]][subgroup_num] < (int)((msg.index - window_size) * num_shard_senders + shard_sender_index))) {
669  std::cout << sst->to_string() << std::endl;
670  return false;
671  }
672  }
673 
674  return true;
675  };
676  auto should_send = [&]() {
677  for(uint i = 1; i <= total_num_subgroups; ++i) {
678  if(should_send_to_subgroup((subgroup_to_send + i) % total_num_subgroups)) {
679  subgroup_to_send = (subgroup_to_send + i) % total_num_subgroups;
680  return true;
681  }
682  }
683  return false;
684  };
685  auto should_wake = [&]() { return thread_shutdown || should_send(); };
686  try {
687  std::unique_lock<std::mutex> lock(msg_state_mtx);
688  while(!thread_shutdown) {
689  sender_cv.wait(lock, should_wake);
690  if(!thread_shutdown) {
691  current_sends[subgroup_to_send] = std::move(pending_sends[subgroup_to_send].front());
692  logger->debug("Calling send in subgroup {} on message {} from sender {}", subgroup_to_send, current_sends[subgroup_to_send]->index, current_sends[subgroup_to_send]->sender_id);
693  if(!rdmc::send(subgroup_to_rdmc_group[subgroup_to_send],
694  current_sends[subgroup_to_send]->message_buffer.mr, 0,
695  current_sends[subgroup_to_send]->size)) {
696  throw std::runtime_error("rdmc::send returned false");
697  }
698  pending_sends[subgroup_to_send].pop();
699  }
700  }
701  std::cout << "DerechoGroup send thread shutting down" << std::endl;
702  } catch(const std::exception& e) {
703  std::cout << "DerechoGroup send thread had an exception: " << e.what() << std::endl;
704  }
705 }

715 char* MulticastGroup::get_sendbuffer_ptr(subgroup_id_t subgroup_num,
716  long long unsigned int payload_size,
717  int pause_sending_turns, bool cooked_send) {
718  // if rdmc groups were not created because of failures, return NULL
719  if(!rdmc_groups_created) {
720  return NULL;
721  }
722  long long unsigned int msg_size = payload_size + sizeof(header);
723  // payload_size is 0 when max_msg_size is desired, useful for ordered send/query
724  if(!payload_size) {
725  msg_size = max_msg_size;
726  }
727  if(msg_size > max_msg_size) {
728  std::cout << "Can't send messages of size larger than the maximum message "
729  "size which is equal to "
730  << max_msg_size << std::endl;
731  return nullptr;
732  }
733 
734  uint32_t shard_num, shard_index;
735  std::tie(shard_num, shard_index) = subgroup_to_shard_and_rank.at(subgroup_num);
736  std::vector<node_id_t> shard_members = subgroup_to_membership.at(subgroup_num);
737  auto num_shard_members = shard_members.size();
738  // if the current node is not a sender, shard_sender_index will be -1
739  uint32_t num_shard_senders;
740  int shard_sender_index;
741  std::vector<int> shard_senders;
742  std::tie(shard_senders, shard_sender_index) = subgroup_to_senders_and_sender_rank.at(subgroup_num);
743  num_shard_senders = get_num_senders(shard_senders);
744  assert(shard_sender_index >= 0);
745 
746  for(uint i = 0; i < num_shard_members; ++i) {
747  if(sst->delivered_num[node_id_to_sst_index[shard_members[i]]][subgroup_num] < (int)((future_message_indices[subgroup_num] - window_size) * num_shard_senders + shard_sender_index)) {
748  return nullptr;
749  }
750  }
751 
752  std::unique_lock<std::mutex> lock(msg_state_mtx);
753  if(thread_shutdown) return nullptr;
754  if(free_message_buffers[subgroup_num].empty()) return nullptr;
755 
756  // Create new Message
757  Message msg;
758  msg.sender_id = members[member_index];
759  msg.index = future_message_indices[subgroup_num];
760  msg.size = msg_size;
761  msg.message_buffer = std::move(free_message_buffers[subgroup_num].back());
762  free_message_buffers[subgroup_num].pop_back();
763 
764  // Fill header
765  char* buf = msg.message_buffer.buffer.get();
766  ((header*)buf)->header_size = sizeof(header);
767  ((header*)buf)->pause_sending_turns = pause_sending_turns;
768  ((header*)buf)->cooked_send = cooked_send;
769 
770  next_sends[subgroup_num] = std::move(msg);
771  future_message_indices[subgroup_num] += pause_sending_turns + 1;
772 
773  return buf + sizeof(header);
774 }
